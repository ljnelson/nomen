028package com.edugility.nomen;
029
030import org.junit.Test;
031
032import static org.junit.Assert.*;
033
034public class TestCaseTemplateEvaluation {
035
036  public TestCaseTemplateEvaluation() {
037    super();
038  }
039
040  @Test
041  public void testEvaluation() {
042    final NameValue laird = new NameValue("Laird", true);
043    assertTrue(laird.isAtomic());
044    assertNull(laird.getWhitespaceReplacement());
045
046    final NameValue jarrett = new NameValue("Jarrett", true);
047    assertTrue(jarrett.isAtomic());
048    assertNull(jarrett.getWhitespaceReplacement());
049
050    final NameValue nelson = new NameValue("Nelson", true);
051    assertTrue(nelson.isAtomic());
052    assertNull(nelson.getWhitespaceReplacement());
053
054    final NameValue lairdJarrettNelson = new NameValue("${firstName}        ${middleName} ${lastName}");
055    assertFalse(lairdJarrettNelson.isAtomic());
056    assertEquals(" ", lairdJarrettNelson.getWhitespaceReplacement());
057
058    final Name firstName = new Name(laird);
059    final Name middleName = new Name(jarrett);
060    final Name lastName = new Name(nelson);
061    final Name fullName = new Name(lairdJarrettNelson);
062
063    // All these Names are initially unowned.
064    assertNull(firstName.getNamed());
065    assertNull(middleName.getNamed());
066    assertNull(lastName.getNamed());
067    assertNull(fullName.getNamed());
068
069    final Named dude = new Named() {
070        private static final long serialVersionUID = 1L;
071        @Override
072        public final Name getName(final NameType nameType) {
073          if (nameType == null) {
074            throw new IllegalArgumentException("nameType", new NullPointerException("nameType"));
075          }
076          final String value = nameType.getValue();
077          if ("firstName".equals(value)) {
078            return firstName;
079          } else if ("middleName".equals(value)) {
080            return middleName;
081          } else if ("lastName".equals(value)) {
082            return lastName;
083          } else if ("fullName".equals(value)) {
084            return fullName;
085          } else {
086            return null;
087          }
088        }
089      };
090
091    // Now that we've got the chicken-and-egg problem out of the way,
092    // "own" all the names.
093    firstName.setNamed(dude);
094    middleName.setNamed(dude);
095    lastName.setNamed(dude);
096    fullName.setNamed(dude);
097
098    // Make sure the ownership "stuck".
099    assertSame(dude, firstName.getNamed());
100    assertSame(dude, middleName.getNamed());
101    assertSame(dude, lastName.getNamed());
102    assertSame(dude, fullName.getNamed());
103
104    final String value = fullName.getValue();
105    assertEquals("Laird Jarrett Nelson", value);
106  }
107
108  @Test(expected = IllegalStateException.class)
109  public void testSyntacticallyInvalidTemplate() {
110    final Name firstName = new Name();
111    final NameValue template = new NameValue("${BAD SYNTAX ON PURPOSE");
112    assertFalse(template.isAtomic());
113
114    // Throws IllegalStateException
115    firstName.setNameValue(template);
116  }
117
118}
