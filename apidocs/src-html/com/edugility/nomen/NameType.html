028    package com.edugility.nomen;
029    
030    import java.io.Serializable;
031    
032    import java.util.LinkedHashMap;
033    import java.util.Map;
034    import java.util.Map.Entry;
035    
036    import java.util.concurrent.locks.ReadWriteLock;
037    import java.util.concurrent.locks.ReentrantReadWriteLock;
038    
039    /**
040     * An {@link AbstractValued} implementation that serves as a key
041     * within a {@link Named} to identify particular {@link Name}
042     * instances.
043     *
044     * @author <a href="http://about.me/lairdnelson"
045     * target="_parent">Laird Nelson</a>
046     *
047     * @see Named
048     *
049     * @see Name
050     *
051     * @see Named#getName(NameType)
052     */
053    public class NameType extends AbstractValued {
054    
055      /**
056       * The version of this class for {@linkplain Serializable
057       * serialization purposes}.
058       *
059       * @see Serializable
060       */
061      private static final long serialVersionUID = 1L;
062    
063      /**
064       * The size of the cache used by the {@link #valueOf(String)}
065       * method; {@code 20} by default and linked to the {@code
066       * nomen.NameType.cacheSize} {@linkplain System#getProperty(String,
067       * String) system property}.
068       *
069       * @see #valueOf(String)
070       */
071      private static final int cacheSize = Integer.getInteger("nomen.NameType.cacheSize", 20);
072      
073      /**
074       * A cache of {@link NameType} instances used by the {@link
075       * #valueOf(String)} method.  A {@linkplain
076       * ReadWriteLock#writeLock() write lock} must be obtained from the
077       * {@link #cacheLock} field before modifying this field, and a
078       * {@linkplain ReadWriteLock#readLock() read lock} must be obtained
079       * from the {@link #cacheLock} field before reading from this field.
080       *
081       * <p>This field is never {@code null}.</p>
082       *
083       * @see #valueOf(String)
084       */
085      private static final Map<String, NameType> cache = new LinkedHashMap<String, NameType>(cacheSize, 0.75F, true) {
086        private static final long serialVersionUID = 1L;
087        @Override
088        protected boolean removeEldestEntry(final Entry<String, NameType> entry) {
089          return this.size() > cacheSize;
090        }
091      };
092      
093      /**
094       * A {@link ReadWriteLock} used to synchronized access to the {@link
095       * #cache} field; used only by the {@link #valueOf(String)} method.
096       *
097       * <p>This field is never {@code null}.</p>
098       */
099      private static final ReadWriteLock cacheLock = new ReentrantReadWriteLock();
100    
101    
102      /*
103       * Constructors.
104       */
105    
106    
107      /**
108       * Creates a new {@link NameType}.
109       */
110      public NameType() {
111        super();
112      }
113    
114      /**
115       * Creates a new {@link NameType} with the supplied {@code value}.
116       *
117       * @param value the value for this {@link NameType}; must not be
118       * {@code null}
119       * 
120       * @exception IllegalArgumentException if {@code value} is {@code
121       * null}
122       *
123       * @see AbstractValued#AbstractValued(String)
124       */
125      public NameType(final String value) {
126        super(value);
127      }
128    
129    
130      /*
131       * Static methods.
132       */
133    
134    
135      /**
136       * Returns the result of calling the {@link #valueOf(String)}
137       * method.  This method exists primarily for using via the {@code
138       * import static} directive.
139       *
140       * <p>This method never returns {@code null}.</p>
141       *
142       * <p>This method is safe for use by multiple concurrent
143       * threads.</p>
144       *
145       * @param value a {@link String} naming the {@link NameType} to be
146       * returned; must not be {@code null}
147       * 
148       * @return the result of calling the {@link #valueOf(String)}
149       * method; never {@code null}
150       *
151       * @exception IllegalArgumentException if {@code value} is {@code
152       * null}
153       *
154       * @see #valueOf(String)
155       *
156       * @see #NameType(String)
157       */
158      public static final NameType nt(final String value) {
159        return valueOf(value);
160      }
161    
162      /**
163       * Returns a {@link NameType} whose {@linkplain #getValue() value}
164       * will be equal to the supplied {@code value}.
165       *
166       * <p>This method never returns {@code null}.</p>
167       *
168       * <p>This method is safe for use by multiple concurrent
169       * threads.</p>
170       *
171       * <p>This method does not necessarily create a new {@link NameType}
172       * on each invocation, but may return a cached copy.</p>
173       *
174       * @param value a {@link String} naming the {@link NameType} to be
175       * returned; must not be {@code null}
176       *
177       * @return a non-{@code null} {@link NameType}
178       *
179       * @exception IllegalArgumentException if {@code value} is {@code
180       * null}
181       *
182       * @see #NameType(String)
183       */
184      public static final NameType valueOf(final String value) {
185        if (value == null) {
186          throw new IllegalArgumentException("value", new NullPointerException("value"));
187        }
188        NameType nt = null;
189        // Because the cache is a LinkedHashMap in access order, get()
190        // calls are structural modifications.  Therefore we need a write
191        // lock.
192        cacheLock.writeLock().lock();
193        try {
194          nt = cache.get(value);
195          if (nt == null) {
196            nt = new NameType(value);
197            cache.put(value, nt);
198          }
199        } finally {
200          cacheLock.writeLock().unlock();
201        }
202        return nt;
203      }
204    
205      /**
206       * Returns {@code true} if and only if the supplied {@code value}
207       * identifies a {@link NameType} that is cached by the internals of
208       * this class.
209       *
210       * <p>The internal cache's contents can change over time, so there
211       * is no guarantee&mdash;explicit or implicit&mdash;about how long a
212       * given cached {@link NameType} will continue to be cached.</p>
213       *
214       * <p>This method is intended for use by single-threaded unit tests,
215       * although it is safe for use by multiple concurrent threads.</p>
216       *
217       * @param value a {@link String} identifying a {@link NameType};
218       * must not be {@code null}
219       *
220       * @return {@code true} if the supplied {@code value} identifies a
221       * cached {@link NameType}; {@code false} otherwise
222       */
223      static final boolean isCached(final String value) {
224        if (value == null) {
225          throw new IllegalArgumentException("value", new NullPointerException("value"));
226        }
227        try {
228          cacheLock.readLock().lock();
229          return cache.containsKey(value);
230        } finally {
231          cacheLock.readLock().unlock();
232        }
233      }
234    
235    }
