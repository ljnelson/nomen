030import java.beans.PropertyChangeListener;
031
032import java.io.IOException;
033import java.io.ObjectInputStream;
034import java.io.Serializable;
035
036import java.util.Collection;
037import java.util.Collections;
038import java.util.HashMap;
039import java.util.Map;
040import java.util.Set;

042/**
043 * An object that manages {@link Name}s on behalf of some {@link
044 * Named}.  A {@link NameSupport} can be used as an internal delegate
045 * in your {@link Named} implementation to handle common use cases.
046 *
047 * @author <a href="http://about.me/lairdnelson"
048 * target="_parent">Laird Nelson</a>
049 */
050public class NameSupport {
051
052  /**
053   * The version of this class for {@linkplain Serializable
054   * serialization purposes}.
055   */
056  private static final long serialVersionUID = 1L;
057
058  /**
059   * The {@link Named} on whose behalf this {@link NameSupport} is
060   * operating.
061   *
062   * <p>This field is never {@code null}.</p>
063   */
064  private final Named delegate;
065
066  /**
067   * Creates a new {@link NameSupport} that will manage {@link Name}s
068   * for the supplied {@link Named}.
069   *
070   * @param delegate the delegate whose {@link Name}s will be managed;
071   * must not be {@code null}
072   *
073   * @exception IllegalArgumentException if {@code delegate} is {@code
074   * null}
075   */
076  public NameSupport(final Named delegate) {
077    super();
078    if (delegate == null) {
079      throw new IllegalArgumentException("delegate", new NullPointerException("delegate"));
080    }
081    this.delegate = delegate;
082  }
083
084  /**
085   * Returns a {@link Set} of {@link NameType}s that the supplied
086   * {@link Map} contains as {@linkplain Map#keySet() keys}.
087   *
088   * <p>This method never returns {@code null}.</p>
089   *
090   * @param map the {@link Map} whose {@linkplain Map#keySet() keys}
091   * are to be returned; may be {@code null} in which case an
092   * {@linkplain Collections#emptySet() empty <code>Set</code>} will
093   * be returned
094   *
095   * @return a non-{@code null}, {@linkplain
096   * Collections#unmodifiableSet(Set) immutable <code>Set</code>}
097   */
098  public Set<? extends NameType> getNameTypes(final Map<? extends NameType, ?> map) {
099    final Set<? extends NameType> returnValue;
100    if (map == null || map.isEmpty()) {
101      returnValue = Collections.emptySet();
102    } else {
103      returnValue = Collections.unmodifiableSet(map.keySet());
104    }
105    return returnValue;
106  }
107
108  /**
109   * Returns a non-{@code null}, {@linkplain
110   * Collections#unmodifiableCollection(Collection) immutable
111   * <code>Collection</code>} of {@link Name}s housed in the supplied
112   * {@link Map} as its {@linkplain Map#values() values}.
113   *
114   * <p>This method never returns {@code null}.</p>
115   *
116   * @param map a {@link Map} containing {@link Name}s as its
117   * {@linkplain Map#values() values}; may be {@code null} in which
118   * case an {@linkplain Collections#emptySet() empty
119   * <code>Set</code>} will be returned
120   *
121   * @return a non-{@code null} {@linkplain
122   * Collections#unmodifiableCollection(Collection) immutable
123   * <code>Collection</code>} of {@link Name}s
124   */
125  public Collection<? extends Name> getNames(final Map<?, ? extends Name> map) {
126    final Collection<Name> returnValue;
127    if (map == null || map.isEmpty()) {
128      returnValue = Collections.emptySet();
129    } else {
130      returnValue = Collections.unmodifiableCollection(map.values());
131    }
132    return returnValue;
133  }
134
135  /**
136   * {@linkplain Map#put(Object, Object) Puts} the supplied {@link
137   * Name} into the supplied {@link Map} indexed under the supplied
138   * {@link NameType}, while ensuring that {@link Name} ownership is
139   * sanely managed.
140   *
141   * <p>This method may return {@code null}.</p>
142   *
143   * @param map a mutable {@link Map} containing {@link Name}s indexed
144   * by {@link NameType}s; must not be {@code null}; this {@link Map}
145   * will be modified as a result of invoking this method
146   *
147   * @param nameType the {@link NameType} under which to index the
148   * supplied {@link Name} in the supplied {@link Map}; must not be
149   * {@code null}
150   *
151   * @param name the {@link Name} to index in the supplied {@link
152   * Map}; must not be {@code null}
153   *
154   * @return the {@link Name} previously indexed under the supplied
155   * {@link NameType}, or {@code null}
156   */
157  public Name putName(final Map<NameType, Name> map, final NameType nameType, final Name name) {
158    if (map == null) {
159      throw new IllegalArgumentException("map", new NullPointerException("map"));
160    } else if (nameType == null) {
161      throw new IllegalArgumentException("nameType", new NullPointerException("nameType"));
162    } else if (name == null) {
163      throw new IllegalArgumentException("name", new NullPointerException("name"));
164    }
165
166    final Named delegate = this.delegate;
167    assert delegate != null;
168
169    final Name old = delegate.getName(nameType);
170    if (old != name) {
171      
172      name.setNamed(delegate);
173      this.addNameOwnershipMonitor(name, map);
174      
175      final Name priorMapValue = map.put(nameType, name);
176      final Collection<? extends Name> names = map.values();
177      this.disown(old, names);
178      if (priorMapValue != old) {
179        this.disown(priorMapValue, names);
180      }
181    }
182
183    return old;
184  }
185
186  /**
187   * Adds a new {@link NameOwnershipMonitor} to the supplied {@link
188   * Name} as a {@link PropertyChangeListener} that listens for
189   * changes in its {@code named} property.  The {@link
190   * NameOwnershipMonitor} is responsible for ensuring that ownership
191   * of the {@link Name} is properly maintained.
192   *
193   * @param name the {@link Name} to monitor; may be {@code null} in
194   * which case no action will be taken
195   *
196   * @param map a {@link Map} whose {@linkplain Map#values() values}
197   * are {@link Name}s; may be {@code null} in which case no action
198   * will be taken
199   */
200  private final void addNameOwnershipMonitor(final Name name, final Map<?, ? extends Name> map) {
201    if (name != null && map != null) {
202      boolean add = true;
203      final PropertyChangeListener[] pcls = name.getPropertyChangeListeners("named");
204      if (pcls != null && pcls.length > 0) {
205        for (final PropertyChangeListener pcl : pcls) {
206          if (pcl instanceof NameOwnershipMonitor) {
207            add = false;
208            break;
209          }
210        }
211      }
212      if (add) {
213        name.addPropertyChangeListener("named", new NameOwnershipMonitor(this.delegate, map.entrySet()));
214      }
215    }
216  }
217
218  /**
219   * Removes any {@link Name} present in the supplied {@link Map} that
220   * is indexed under the supplied {@link NameType}.
221   *
222   * <p>This method may return {@code null}.</p>
223   *
224   * @param map the {@link Map} from which a {@link Name} should be
225   * removed; may be {@code null} in which case no action will be
226   * taken and {@code null} will be returned
227   *
228   * @param nameType the {@link NameType} identifying the {@link Name}
229   * to be removed; must not be {@code null}
230   *
231   * @return the {@link Name} that was removed, or {@code null}
232   *
233   * @exception IllegalArgumentException if {@code nameType} is {@code
234   * null}
235   */
236  public Name removeName(final Map<?, ? extends Name> map, final NameType nameType) {
237    if (nameType == null) {
238      throw new IllegalArgumentException("nameType", new NullPointerException("nameType"));
239    }
240
241    final Name returnValue;
242    if (map != null && !map.isEmpty()) {
243      returnValue = map.remove(nameType);
244      if (returnValue != null) {
245        this.disown(returnValue, map.values());
246      }
247    } else {
248      returnValue = null;
249    }
250    return returnValue;
251  }
252
253  /**
254   * Sets the ownership of the supplied {@link Name} to {@code null}
255   * if and only if it can be proved that no {@link NameType} indexes
256   * it.
257   *
258   * @param name the {@link Name} to disown; may be {@code null} in
259   * which case no action will be taken
260   *
261   * @param names a {@link Collection} of {@link Name}s that will be
262   * consulted for ownership maintenance purposes; may be {@code null}
263   * in which case no action will be taken
264   */
265  private final void disown(final Name name, final Collection<? extends Name> names) {
266    if (name != null && name.getNamed() != null && names != null && !names.isEmpty()) {
267      boolean found = false;
268      for (final Object value : names) {
269        if (value == name) {
270          found = true;
271          break;
272        }
273      }
274      if (!found) {
275        name.setNamed(null);
276      }
277    }
278  }
