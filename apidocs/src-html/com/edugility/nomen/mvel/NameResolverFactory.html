046/**
047 * A {@link BaseVariableResolverFactory} that produces {@link
048 * NameResolver} instances for use by <a
049 * href="http://mvel.codehaus.org/">MVEL</a> expressions.
050 *
051 * @author <a href="http://about.me/lairdnelson"
052 * target="_parent">Laird Nelson</a>
053 *
054 * @see NameResolver
055 *
056 * @see <a href="http://mvel.codehaus.org/">MVEL</a>
057 *
058 * @see BaseVariableResolverFactory
059 *
060 * @see VariableResolverFactory
061 *
062 * @see VariableResolver
063 *
064 * @see Name#getValue()
065 *
066 * @see NameValue#setAtomic(boolean)
067 */
068public class NameResolverFactory extends BaseVariableResolverFactory implements Named {
069
070  
071  /*
072   * Static fields.
073   */
074
075
076  /**
077   * The version of this class for {@linkplain Serializable
078   * serialization purposes}.
079   *
080   * @see Serializable
081   */
082  private static final long serialVersionUID = 1L;
083
084
085  /*
086   * Instance fields.
087   */
088
089
090  /**
091   * The {@link Named} that serves as the virtual universe or domain
092   * of {@link Name}s that are potentially {@linkplain
093   * #isResolveable(String) resolvable} into {@link NameResolver}s
094   * vended by this {@link NameResolverFactory}.
095   *
096   * <p>This field is never {@code null}.</p>
097   */
098  private final Named named;
099
100
101  /*
102   * Constructors.
103   */
104
105
106  /**
107   * Creates a new {@link NameResolverFactory} given a non-{@code
108   * null} {@link Named} that serves as the universe or domain from
109   * which {@link Name}s may ultimately be resolved.
110   *
111   * @param named the {@link Named} that serves as the universe or
112   * domain from which {@link Name}s may ultimately be resolved; must
113   * not be {@code null}
114   *
115   * @exception IllegalArgumentException if {@code named} is {@code
116   * null}
117   */
118  public NameResolverFactory(final Named named) {
119    super();
120    if (named == null) {
121      throw new IllegalStateException("named", new NullPointerException("named"));
122    }
123    this.named = named;
124  }
125
126  @Override
127  public final Name getName(final NameType nameType) {
128    final Name returnValue;
129    if (nameType != null && this.named != null) {
130      returnValue = this.named.getName(nameType);
131    } else {
132      returnValue = null;
133    }
134    return returnValue;
135  }
136
137  /**
138   * Returns {@code true} if {@code name} is equal to the {@linkplain
139   * NameType#getValue() value} of a {@link NameType} that, when
140   * passed to {@link Named#getName(NameType)}, results in a
141   * non-{@code null} {@link Name}, or if subsequent {@link
142   * VariableResolver}s are able to make sense of the supplied {@code
143   * name} parameter.
144   *
145   * @param name the {@linkplain NameType#getValue() value} of a
146   * {@link NameType}; may be {@code null}
147   *
148   * @return {@code true} if this {@link NameResolverFactory} or any
149   * of its chained {@link VariableResolverFactory} instances can
150   * resolve an MVEL variable of the given name; {@code false}
151   * otherwise
152   *
153   * @see VariableResolverFactory
154   *
155   * @see BaseVariableResolverFactory#isNextResolveable(String)
156   *
157   * @see VariableResolver
158   *
159   * @see NameResolver
160   */
161  @Override
162  public boolean isResolveable(final String name) {
163    final boolean returnValue;
164    if (name == null) {
165      returnValue = this.isNextResolveable(null);
166    } else if (this.isTarget(name)) {
167      returnValue = true;
168    } else {
169      final Name n = this.getName(new NameType(name));
170      if (n == null) {
171        returnValue = this.isNextResolveable(null);
172      } else {
173        if (this.variableResolvers == null) {
174          this.variableResolvers = new HashMap<String, VariableResolver>();
175        }
176        this.variableResolvers.put(name, this.createNameResolver(this.named, name));
177        returnValue = true;
178      }
179    }
180    return returnValue;
181  }
182
183  /**
184   * Creates and returns a {@link NameResolver} appropriate for the
185   * supplied {@link Named} and the supplied {@code value}.
186   *
187   * @param named the {@link Named} that serves as the virtual
188   * universe or domain of {@link Name}s that are potentially
189   * {@linkplain #isResolveable(String) resolvable} into {@link
190   * NameResolver}s vended by this {@link NameResolverFactory}; must
191   * not be {@code null}
192   *
193   * @param name the name of the variable the new {@link NameResolver}
194   * will return from its {@link NameResolver#getName()} method; must
195   * not be {@code null}
196   *
197   * @return a non-{@code null} {@link NameResolver}
198   *
199   * @exception IllegalArgumentException if either {@code named} or
200   * {@code name} is {@code null}
201   */
202  protected NameResolver createNameResolver(final Named named, final String name) {
203    return new NameResolver(named, new NameType(name));
204  }
205
206  /**
207   * Returns {@code true} if the supplied {@code name} is non-{@code
208   * null} and this {@link NameResolverFactory} has a {@link
209   * NameResolver} that has been installed by the inner workings of
210   * the {@link #isResolveable(String)} method to resolve a {@link
211   * Name} indexed under the supplied {@code name}.
212   *
213   * @param name the {@linkplain NameType#getValue() value} of a
214   * {@link NameType}; may be {@code null}
215   *
216   * @return {@code true} if the supplied {@code name} is non-{@code
217   * null} and this {@link NameResolverFactory} has a {@link
218   * NameResolver} that has been installed by the inner workings of
219   * the {@link #isResolveable(String)} method to resolve a {@link
220   * Name} indexed under the supplied {@code name}; {@code false} in
221   * all other cases
222   */
223  @Override
224  public boolean isTarget(final String name) {
225    return name != null && this.variableResolvers != null && this.variableResolvers.containsKey(name);
226  }
227
228  /**
229   * Throws an {@link UnsupportedOperationException} when invoked.
230   *
231   * @param name the name of a variable to create; may be {@code
232   * null}; ignored by this implementation
233   *
234   * @param value the value of the new variable; may be {@code null};
235   * ignored by this implementation
236   *
237   * @return a {@link VariableResolver} responsible for resolving the
238   * new variable, or {@code null}
239   *
240   * @exception UnsupportedOperationException if this implementation
241   * is invoked
242   */
243  @Override
244  public VariableResolver createVariable(final String name, final Object value) {
245    throw new UnsupportedOperationException("createVariable");
246  }
247
248  /**
249   * Throws an {@link UnsupportedOperationException} when invoked.
250   *
251   * @param name the name of a variable to create; may be {@code
252   * null}; ignored by this implementation
253   *
254   * @param value the value of the new variable; may be {@code null};
255   * ignored by this implementation
256   *
257   * @param type the type of the new variable; may be {@code null};
258   * ignored by this implementation
259   *
260   * @return a {@link VariableResolver} responsible for resolving the
261   * new variable, or {@code null}
262   *
263   * @exception UnsupportedOperationException if this implementation
264   * is invoked
265   */
266  @Override
267  public VariableResolver createVariable(final String name, final Object value, final Class<?> type) {
268    throw new UnsupportedOperationException("createVariable");
269  }
