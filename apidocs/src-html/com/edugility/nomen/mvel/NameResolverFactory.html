049/**
050 * A {@link BaseVariableResolverFactory} that produces {@link
051 * NameResolver} instances for use by <a
052 * href="http://mvel.codehaus.org/">MVEL</a> expressions.

054 * <p>Subclasses of this class normally have to override only the

057 * @author <a href="http://about.me/lairdnelson"

140  /**
141   * Given a {@link NameType}, returns a {@link Name} corresponding to
142   * it in some fashion.
143   *
144   * <p>This method may return {@code null}.</p>
145   *
146   * <p>The default implementation of this method invokes the {@link
147   * Named#getName(NameType)} method on the value of the {@link
148   * #named} field.</p>
149   *
150   * @param nameType the {@link NameType} for which a corresponding
151   * {@link Name} should be returned; may be {@code null}
152   *
153   * @return a {@link Name} corresponding to the supplied {@link
154   * NameType}, or {@code null}
155   *
156   * @see #named
157   *
158   * @see Named#getName(NameType)
159   */
160  @Override
161  public Name getName(final NameType nameType) {
162    final Name returnValue;
163    if (nameType != null && this.named != null) {
164      returnValue = this.named.getName(nameType);
165    } else {
166      returnValue = null;
167    }
168    return returnValue;
169  }

171  /**
172   * Returns {@code true} if {@code name} is equal to the {@linkplain
173   * NameType#getValue() value} of a {@link NameType} that, when
174   * passed to {@link Named#getName(NameType)}, results in a
175   * non-{@code null} {@link Name}, or if subsequent {@link
176   * VariableResolver}s are able to make sense of the supplied {@code
177   * name} parameter.
178   *
179   * @param name the {@linkplain NameType#getValue() value} of a
180   * {@link NameType}; may be {@code null}
181   *
182   * @return {@code true} if this {@link NameResolverFactory} or any
183   * of its chained {@link VariableResolverFactory} instances can
184   * resolve an MVEL variable of the given name; {@code false}
185   * otherwise
186   *
187   * @see VariableResolverFactory
188   *
189   * @see BaseVariableResolverFactory#isNextResolveable(String)
190   *
191   * @see VariableResolver
192   *
193   * @see NameResolver
194   */
195  @Override
196  public boolean isResolveable(final String name) {
197    final boolean returnValue;
198    if (name == null) {
199      returnValue = this.isNextResolveable(null);
200    } else if (this.isTarget(name)) {
201      returnValue = true;
202    } else {
203      final Name n = this.getName(new NameType(name));
204      if (n == null) {
205        returnValue = this.isNextResolveable(name);
206      } else {
207        Map<String, VariableResolver> resolvers = this.getVariableResolvers();
208        if (resolvers == null) {
209          resolvers = new HashMap<String, VariableResolver>();
210        }
211        resolvers.put(name, this.createNameResolver(this.named, name));
212        if (resolvers != this.getVariableResolvers()) {
213          this.setVariableResolvers(resolvers);
214        }
215        returnValue = true;
216      }
217    }
218    return returnValue;
219  }
220
221  /**
222   * Overrides the {@link
223   * BaseVariableResolverFactory#getVariableResolver(String)} method
224   * for performance and robustness only.
225   *
226   * <p>This method never returns {@code null}.</p>
227   *
228   * @param name the name of the variable whose {@link
229   * VariableResolver} should be returned, if possible; may be {@code
230   * null}
231   *
232   * @return a {@link VariableResolver} suitable for resolving a
233   * variable with the supplied {@code name}; never {@code null}
234   *
235   * @exception UnresolveablePropertyException if no {@link
236   * VariableResolver} could be found
237   *
238   * @see BaseVariableResolverFactory#getVariableResolver(String)
239   */
240  @Override
241  public final VariableResolver getVariableResolver(final String name) {
242    VariableResolver returnValue = null;
243    if (this.isResolveable(name)) {
244      final Map<String, VariableResolver> resolvers = this.getVariableResolvers();
245      if (resolvers != null && !resolvers.isEmpty()) {
246        returnValue = resolvers.get(name);
247        if (returnValue == null && this.isNextResolveable(name)) {
248          final VariableResolverFactory nextFactory = this.getNextFactory();
249          if (nextFactory != null) {
250            returnValue = nextFactory.getVariableResolver(name);
251          }
252        }
253      }
254    }
255    if (returnValue == null) {
256      throw new UnresolveablePropertyException("unable to resolve variable '" + name + "'");
257    }
258    return returnValue;
259  }

261  /**
262   * Creates and returns a {@link NameResolver} appropriate for the
263   * supplied {@link Named} and the supplied {@code value}.
264   *
265   * @param named the {@link Named} that serves as the virtual
266   * universe or domain of {@link Name}s that are potentially
267   * {@linkplain #isResolveable(String) resolvable} into {@link
268   * NameResolver}s vended by this {@link NameResolverFactory}; must
269   * not be {@code null}
270   *
271   * @param name the name of the variable the new {@link NameResolver}
272   * will return from its {@link NameResolver#getName()} method; must
273   * not be {@code null}
274   *
275   * @return a non-{@code null} {@link NameResolver}
276   *
277   * @exception IllegalArgumentException if either {@code named} or
278   * {@code name} is {@code null}
279   */
280  protected NameResolver createNameResolver(final Named named, final String name) {
281    return new NameResolver(named, new NameType(name));
282  }
283
284  /**
285   * Returns {@code true} if the supplied {@code name} is non-{@code
286   * null} and this {@link NameResolverFactory} has a {@link
287   * NameResolver} that has been installed by the inner workings of
288   * the {@link #isResolveable(String)} method to resolve a {@link
289   * Name} indexed under the supplied {@code name}.
290   *
291   * @param name the {@linkplain NameType#getValue() value} of a
292   * {@link NameType}; may be {@code null}
293   *
294   * @return {@code true} if the supplied {@code name} is non-{@code
295   * null} and this {@link NameResolverFactory} has a {@link
296   * NameResolver} that has been installed by the inner workings of
297   * the {@link #isResolveable(String)} method to resolve a {@link
298   * Name} indexed under the supplied {@code name}; {@code false} in
299   * all other cases
300   */
301  @Override
302  public boolean isTarget(final String name) {
303    final boolean returnValue;
304    if (name == null) {
305      returnValue = false;
306    } else {
307      final Map<?, ?> resolvers = this.getVariableResolvers();
308      returnValue = resolvers != null && resolvers.containsKey(name);
309    }
310    return returnValue;
311  }
