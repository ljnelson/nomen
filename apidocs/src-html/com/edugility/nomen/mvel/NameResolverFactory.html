039    import org.mvel2.integration.VariableResolver;

042    import org.mvel2.integration.impl.BaseVariableResolverFactory;

044    /**
045     * A {@link BaseVariableResolverFactory} that produces {@link
046     * NameResolver} instances for use by <a
047     * href="http://mvel.codehaus.org/">MVEL</a> expressions.
048     *
049     * @author <a href="http://about.me/lairdnelson"
050     * target="_parent">Laird Nelson</a>
051     *
052     * @see NameResolver
053     *
054     * @see <a href="http://mvel.codehaus.org/">MVEL</a>
055     *
056     * @see BaseVariableResolverFactory
057     *
058     * @see VariableResolverFactory
059     *
060     * @see VariableResolver
061     *
062     * @see Name#getValue()
063     *
064     * @see NameValue#setAtomic(boolean)
065     */
066    public class NameResolverFactory extends BaseVariableResolverFactory implements Named {
067    
068      
069      /*
070       * Static fields.
071       */
072    
073    
074      /**
075       * The version of this class for {@linkplain Serializable
076       * serialization purposes}.
077       *
078       * @see Serializable
079       */
080      private static final long serialVersionUID = 1L;
081    
082    
083      /*
084       * Instance fields.
085       */
086    
087    
088      /**
089       * The {@link Named} that serves as the virtual universe or domain
090       * of {@link Name}s that are potentially {@linkplain
091       * #isResolveable(String) resolvable} into {@link NameResolver}s
092       * vended by this {@link NameResolverFactory}.
093       *
094       * <p>This field is never {@code null}.</p>
095       */
096      private final Named named;
097    
098    
099      /*
100       * Constructors.
101       */
102    
103    
104      /**
105       * Creates a new {@link NameResolverFactory} given a non-{@code
106       * null} {@link Named} that serves as the universe or domain from
107       * which {@link Name}s may ultimately be resolved.
108       *
109       * @param named the {@link Named} that serves as the universe or
110       * domain from which {@link Name}s may ultimately be resolved; must
111       * not be {@code null}
112       *
113       * @exception IllegalArgumentException if {@code named} is {@code
114       * null}
115       */
116      public NameResolverFactory(final Named named) {
117        super();
118        if (named == null) {
119          throw new IllegalStateException("named", new NullPointerException("named"));
120        }
121        this.named = named;
122      }
123    
124      @Override
125      public final Name getName(final NameType nameType) {
126        final Name returnValue;
127        if (nameType != null && this.named != null) {
128          returnValue = this.named.getName(nameType);
129        } else {
130          returnValue = null;
131        }
132        return returnValue;
133      }
134      
135      /**
136       * Returns {@code true} if {@code name} is equal to the {@linkplain
137       * NameType#getValue() value} of a {@link NameType} that, when
138       * passed to {@link Named#getName(NameType)}, results in a
139       * non-{@code null} {@link Name}, or if subsequent {@link
140       * VariableResolver}s are able to make sense of the supplied {@code
141       * name} parameter.
142       *
143       * @param name the {@linkplain NameType#getValue() value} of a
144       * {@link NameType}; may be {@code null}
145       *
146       * @return {@code true} if this {@link NameResolverFactory} or any
147       * of its chained {@link VariableResolverFactory} instances can
148       * resolve an MVEL variable of the given name; {@code false}
149       * otherwise
150       *
151       * @see VariableResolverFactory
152       *
153       * @see BaseVariableResolverFactory#isNextResolveable(String)
154       *
155       * @see VariableResolver
156       *
157       * @see NameResolver
158       */
159      @Override
160      public boolean isResolveable(final String name) {
161        final boolean returnValue;
162        if (name == null) {
163          returnValue = this.isNextResolveable(null);
164        } else if (this.isTarget(name)) {
165          returnValue = true;
166        } else {
167          final Name n = this.getName(NameType.valueOf(name));
168          if (n == null) {
169            returnValue = this.isNextResolveable(null);
170          } else {
171            if (this.variableResolvers == null) {
172              this.variableResolvers = new HashMap<String, VariableResolver>();
173            }
174            this.variableResolvers.put(name, this.createNameResolver(this.named, name));
175            returnValue = true;
176          }
177        }
178        return returnValue;
179      }
180    
181      /**
182       * Creates and returns a {@link NameResolver} appropriate for the
183       * supplied {@link Named} and the supplied {@code value}.
184       *
185       * @param named the {@link Named} that serves as the virtual
186       * universe or domain of {@link Name}s that are potentially
187       * {@linkplain #isResolveable(String) resolvable} into {@link
188       * NameResolver}s vended by this {@link NameResolverFactory}; must
189       * not be {@code null}
190       *
191       * @param name the name of the variable the new {@link NameResolver}
192       * will return from its {@link NameResolver#getName()} method; must
193       * not be {@code null}
194       *
195       * @return a non-{@code null} {@link NameResolver}
196       *
197       * @exception IllegalArgumentException if either {@code named} or
198       * {@code name} is {@code null}
199       */
200      protected NameResolver createNameResolver(final Named named, final String name) {
201        return new NameResolver(named, NameType.valueOf(name));
202      }
203    
204      /**
205       * Returns {@code true} if the supplied {@code name} is non-{@code
206       * null} and this {@link NameResolverFactory} has a {@link
207       * NameResolver} that has been installed by the inner workings of
208       * the {@link #isResolveable(String)} method to resolve a {@link
209       * Name} indexed under the supplied {@code name}.
210       *
211       * @param name the {@linkplain NameType#getValue() value} of a
212       * {@link NameType}; may be {@code null}
213       *
214       * @return {@code true} if the supplied {@code name} is non-{@code
215       * null} and this {@link NameResolverFactory} has a {@link
216       * NameResolver} that has been installed by the inner workings of
217       * the {@link #isResolveable(String)} method to resolve a {@link
218       * Name} indexed under the supplied {@code name}; {@code false} in
219       * all other cases
220       */
221      @Override
222      public boolean isTarget(final String name) {
223        return name != null && this.variableResolvers != null && this.variableResolvers.containsKey(name);
224      }
225    
226      /**
227       * Throws an {@link UnsupportedOperationException} when invoked.
228       *
229       * @param name the name of a variable to create; may be {@code
230       * null}; ignored by this implementation
231       *
232       * @param value the value of the new variable; may be {@code null};
233       * ignored by this implementation
234       *
235       * @return a {@link VariableResolver} responsible for resolving the
236       * new variable, or {@code null}
237       *
238       * @exception UnsupportedOperationException if this implementation
239       * is invoked
240       */
241      @Override
242      public VariableResolver createVariable(final String name, final Object value) {
243        throw new UnsupportedOperationException("createVariable");
244      }
245    
246      /**
247       * Throws an {@link UnsupportedOperationException} when invoked.
248       *
249       * @param name the name of a variable to create; may be {@code
250       * null}; ignored by this implementation
251       *
252       * @param value the value of the new variable; may be {@code null};
253       * ignored by this implementation
254       *
255       * @param type the type of the new variable; may be {@code null};
256       * ignored by this implementation
257       *
258       * @return a {@link VariableResolver} responsible for resolving the
259       * new variable, or {@code null}
260       *
261       * @exception UnsupportedOperationException if this implementation
262       * is invoked
263       */
264      @Override
265      public VariableResolver createVariable(final String name, final Object value, final Class<?> type) {
266        throw new UnsupportedOperationException("createVariable");
267      }
268
