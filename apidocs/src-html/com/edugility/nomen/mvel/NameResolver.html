028    package com.edugility.nomen.mvel;
029    
030    import com.edugility.nomen.AbstractNamed;
031    import com.edugility.nomen.Name;
032    import com.edugility.nomen.Named;
033    import com.edugility.nomen.NameType;
034    import com.edugility.nomen.NameValue;
035    
036    import org.mvel2.integration.VariableResolver;
037    
038    import org.mvel2.integration.impl.BaseVariableResolverFactory;
039    import org.mvel2.integration.impl.SimpleValueResolver;

041    /**
042     * An <a href="http://mvel.codehaus.org/">MVEL</a> {@link
043     * VariableResolver} for resolving MVEL variables from a {@link
044     * Named}.
045     *
046     * @author <a href="http://about.me/lairdnelson"
047     * target="_parent">Laird Nelson</a>
048     *
049     * @see NameResolverFactory
050     *
051     * @see VariableResolver
052     *
053     * @see <a href="http://mvel.codehaus.org">MVEL</a>
054     */
055    public final class NameResolver implements VariableResolver, Named {
056    
057      
058      /*
059       * Static fields.
060       */
061    
062    
063      /**
064       * The version of this class for {@linkplain Serializable
065       * serialization purposes}.
066       *
067       * @see Serializable
068       */
069      private static final long serialVersionUID = 1L;
070    
071    
072      /*
073       * Instance fields.
074       */
075    
076    
077      /**
078       * The {@link Named} that owns {@link Name}s that might be resolved
079       * by this {@link NameResolver}.
080       *
081       * <p>This field will never be {@code null}.</p>
082       *
083       * @see Named
084       */
085      private final Named named;
086    
087      /**
088       * The {@link NameType} used by this {@link NameResolver} to {@link
089       * #getName(NameType) look up <code>Name</code>s}.
090       *
091       * <p>This field will never be {@code null}.</p>
092       *
093       * @see #getName(NameType)
094       *
095       * @see NameType
096       */
097      private final NameType nameType;
098    
099    
100      /*
101       * Constructors.
102       */
103    
104    
105      /**
106       * Creates a new {@link NameResolver}.
107       *
108       * @param named the {@link Named} that will be used by the {@link
109       * #getValue()} method; must not be {@code null}
110       *
111       * @param nameType the {@link NameType} with which this {@link
112       * NameResolver} will be affiliated; must not be {@code null}
113       *
114       * @exception IllegalArgumentException if either {@code named} or
115       * {@code nameType} is {@code null}
116       */
117      public NameResolver(final Named named, final NameType nameType) {
118        super();
119        if (named == null) {
120          throw new IllegalArgumentException("named", new NullPointerException("named"));
121        }
122        if (nameType == null) {
123          throw new IllegalArgumentException("nameType", new NullPointerException("nameType"));
124        }
125        this.named = named;
126        this.nameType = nameType;
127      }
128    
129    
130      /*
131       * Instance methods.
132       */
133    
134    
135      /**
136       * Returns {@code 0}, following the recommendations of the <a
137       * href="http://mvel.codehaus.org/">MVEL</a> project.
138       *
139       * <p>The {@link VariableResolver} documentation does not describe
140       * what this method is used for.</p>
141       *
142       * @return {@code 0}
143       */
144      @Override
145      public final int getFlags() {
146        return 0; // per VariableResolver "documentation"
147      }
148    
149      /**
150       * Returns the {@link String} that is the {@linkplain
151       * NameType#getValue() value} of the {@link NameType} supplied at
152       * {@linkplain #NameResolver(Named, NameType) construction time}.
153       *
154       * <p>This method never returns {@code null}.</p>
155       *
156       * @return a non-{@code null} {@link String} naming the {@link Name}
157       * whose value will be returned by the {@link #getValue()} method
158       *
159       * @see #getValue()
160       */
161      @Override
162      public final String getName() {
163        assert this.nameType != null;
164        return this.nameType.getValue();
165      }
166    
167      /**
168       * Returns a {@link Name} for the supplied {@link NameType}, or
169       * {@code null} if no such {@link Name} could be found.
170       *
171       * @param nameType the {@link NameType} to use as a key; may be
172       * {@code null}
173       *
174       * @return a {@link Name}, or {@code null}
175       *
176       * @see Named#getName(NameType)
177       */
178      @Override
179      public final Name getName(final NameType nameType) {
180        assert this.named != null;
181        return this.named.getName(nameType);
182      }
183    
184      /**
185       * Returns the value of {@code Name.class} when invoked.
186       *
187       * <p>This method never returns {@code null}.</p>
188       *
189       * @return {@code Name.class}
190       */
191      @Override
192      public final Class<Name> getType() {
193        return Name.class;
194      }
195    
196      /**
197       * Returns the value of a {@link Name} reference in a <a
198       * href="http://mvel.codehaus.org/">MVEL</a> program whose name is
199       * given by the return value of this {@link NameResolver}'s {@link
200       * #getName()} method.
201       *
202       * <p>If a suitable value cannot be determined, then the {@linkplain
203       * String#isEmpty() empty string} ("") is returned.</p>
204       *
205       * <p>This method never returns {@code null}.</p>
206       *
207       * @return a non-{@code null} {@link String}
208       *
209       * @see #getName()
210       */
211      @Override
212      public final Object getValue() {
213        Object returnValue = null;
214        final Name n = this.getName(this.nameType);
215        if (n != null) {
216          returnValue = n.getValue();
217        }
218        if (returnValue == null) {
219          returnValue = "";
220        }
221        return returnValue;
222      }
223    
224      /**
225       * Does nothing.
226       *
227       * @param type ignored
228       *
229       * @see #getType()
230       */
231      @Override
232      @SuppressWarnings("rawtypes")
233      public final void setStaticType(final Class type) {
234    
235      }
236    
237      /**
238       * <strong>Experimental</strong>; {@linkplain
239       * AbstractNamed#putName(NameType, NameValue) puts} a {@link Name}
240       * into the {@link Named} that is affiliated with this {@link
241       * NameResolver}, provided that it is an instance of {@link
242       * AbstractNamed}.
243       *
244       * <p>In general, if the conditions described in the parameters
245       * section below are not met, this method does nothing.</p>
246       *
247       * @param value the value for a new name; may be {@code null}, a
248       * {@link NameValue}, a {@link String} or a {@link Name}.  If it is
249       * {@code null}, then any {@link Name} indexed under this {@link
250       * NameResolver}'s affiliated {@link NameType} present in the
251       * affiliated {@link AbstractNamed} will be removed.  If it is
252       * either a {@link NameValue}, a {@link String} or a {@link Name},
253       * then it is converted common-sensically into a {@link NameValue}
254       * and is {@linkplain AbstractNamed#putName(NameType, NameValue)
255       * installed into} the {@link AbstractNamed} in such a way that a
256       * subsequent call to {@link #getValue()} will return the expected
257       * result.
258       */
259      @Override
260      public final void setValue(final Object value) {
261        if (this.named instanceof AbstractNamed) {
262          final AbstractNamed owner = (AbstractNamed)this.named;
263          if (value == null) {
264            // remove
265            owner.removeName(this.nameType);
266          } else {
267            NameValue nv = null;
268            if (value instanceof NameValue) {
269              nv = (NameValue)value;
270            } else if (value instanceof String) {
271              nv = NameValue.valueOf((String)value, false, " ");
272            } else if (value instanceof Name) {
273              nv = ((Name)value).getNameValue();
274            } else {
275              nv = null;
276            }
277            if (nv != null) {
278              owner.putName(this.nameType, nv);
279            }
280          }
281        }
282      }
