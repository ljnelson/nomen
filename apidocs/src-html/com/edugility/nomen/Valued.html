028package com.edugility.nomen;
029
030/**
031 * Indicates that an implementation is capable of reporting its value
032 * as a {@link String}.
033 *
034 * @author <a href="http://about.me/lairdnelson"
035 * target="_parent">Laird Nelson</a>
036 *
037 * @see #getValue()
038 */
039public interface Valued {
040
041  /**
042   * Returns the value of this {@link Valued} implementation as a
043   * {@link String}.
044   *
045   * <p>Implementations of this method must not return {@code
046   * null}.</p>
047   *
048   * @return a non-{@code null} {@link String}
049   */
050  public String getValue();
051
052}
