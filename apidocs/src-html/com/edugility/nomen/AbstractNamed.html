028package com.edugility.nomen;
029
030import java.io.Serializable;
031
032import java.util.Collection;
033import java.util.Collections;
034import java.util.HashMap;
035import java.util.Map;
036import java.util.Set;
037
038/**
039 * A skeletal implementation of the {@link Named} interface, together
040 * with mutator methods.
041 *
042 * @author <a href="http://about.me/lairdnelson"
043 * target="_parent">Laird Nelson</a>
044 */
045public abstract class AbstractNamed implements Named {
046
047
048  /*
049   * Static fields.
050   */
051
052
053  /**
054   * The version of this class for {@linkplain Serializable
055   * serialization purposes}.
056   */
057  private static final long serialVersionUID = 1L;
058
059
060  /*
061   * Instance fields.
062   */
063
064
065  /**
066   * A {@link Map} of the {@link Name}s this {@link AbstractNamed}
067   * has, indexed by {@link NameType}.
068   *
069   * <p>This field may be {@code null}.</p>
070   */
071  private Map<NameType, Name> names;
072
073
074  /*
075   * Constructors.
076   */
077
078
079  /**
080   * Creates a new {@link AbstractNamed}.
081   */
082  protected AbstractNamed() {
083    super();
084  }
085
086
087  /*
088   * Instance methods.
089   */
090
091
092  /**
093   * Retrieves a {@link Name} that belongs to this {@link
094   * AbstractNamed} that is appropriate for the supplied {@link
095   * NameType}, or {@code null} if no such {@link Name} can be found.
096   *
097   * <p>This method may return {@code null}.</p>
098   *
099   * @param nameType the {@link NameType} for which a {@link Name}
100   * will be sought; may be {@code null} in which case {@code null}
101   * will be returned
102   *
103   * @return a {@link Name} indexed under the supplied {@link
104   * NameType}, or {@code null}
105   *
106   * @exception IllegalArgumentException if {@code nameType} is not
107   * accepted for any reason
108   */
109  @Override
110  public Name getName(final NameType nameType) {
111    Name name = null;
112    if (nameType != null && this.names != null && !this.names.isEmpty()) {
113      name = this.names.get(nameType);
114      assert name == null ? true : name.getNamed() == this;
115    }
116    return name;
117  }
118
119  /**
120   * Stores the supplied {@link Name} under the supplied {@link
121   * NameType} in such a way that it can be retrieved later by the
122   * #getName(NameType)} method when that method is supplied with a
123   * {@link NameType} {@linkplain NameType#equals(Object) equal to}
124   * the supplied {@link NameType}.
125   *
126   * <h3>Implementation Notes</h3>
127   *
128   * <p>To properly orphan {@link Name}s that are no longer indexed
129   * under a given {@link NameType} in this {@link AbstractNamed},
130   * {@linkplain #getNames() all the <code>Name</code>s affiliated
131   * with this <code>AbstractNamed</code> implementation} must be
132   * scanned, so this operation has performance that decreases as
133   * the total number of {@link Name}s stored by this {@link
134   * AbstractNamed} implementation increases.</p>
135   *
136   * @param nameType a {@link NameType} under which a new {@link Name}
137   * will be stored; must not be {@code null}
138   *
139   * @param name a {@link Name} to store; must not be {@code null};
140   * its {@link Name#setNamed(Named)} method will be called with
141   * {@code this} as its argument.
142   *
143   * @return the prior {@link Name} stored under the supplied {@link
144   * NameType}, or {@code null} if there was no such {@link Name}.
145   * The returned {@link Name}, if non-{@code null}, will return
146   * {@code null} from its {@link Name#getNamed()} method only if it
147   * is now truly orphaned, that is only if it is not indexed in this
148   * {@link AbstractNamed} under any other {@link NameType}.
149   *
150   * @exception IllegalArgumentException if {@code nameType} or {@code
151   * name} is {@code null}, or if {@code nameType} is unsuitable, or
152   * if {@code name} is unsuitable
153   */
154  public Name putName(final NameType nameType, final Name name) {
155    if (nameType == null) {
156      throw new IllegalArgumentException("nameType", new NullPointerException("nameType"));
157    } else if (name == null) {
158      throw new IllegalArgumentException("name", new NullPointerException("name"));
159    }
160    if (this.names == null) {
161      this.names = new HashMap<NameType, Name>(11);
162    }
163    final Name returnValue;
164    final Name old = this.names.get(nameType);
165    if (old == name) {
166      returnValue = old;
167    } else {
168      name.setNamed(this);
169      returnValue = this.names.put(nameType, name);
170      assert returnValue == old;
171      if (returnValue != null) {
172        final Collection<?> values = this.names.values();
173        if (values != null && !values.isEmpty()) {
174          boolean found = false;
175          for (final Object value : values) {
176            if (value == returnValue) {
177              // Can't use equals() here, because two Names are equal
178              // if their computed values result in the same string.
179              // Here we are truly interested in object identity, not
180              // equality.
181              found = true;
182              break;
183            }
184          }
185          if (!found) {
186            // This returnValue is not stored anywhere else in the
187            // map, so we can orphan it.
188            returnValue.setNamed(null);
189          }
190        }
191      }
192    }
193    return returnValue;
194  }
195
196  /**
197   * Creates a new {@link Name} implementation given a non-{@code
198   * null} {@link NameValue}.
199   *
200   * <p>This implementation simply {@linkplain Name#Name(Named,
201   * NameValue) calls the relevant <code>Name</code> constructor}.</p>
202   *
203   * @param nameValue the {@link NameValue} that will serve as the
204   * value for the new {@link Name}; must not be {@code null}
205   *
206   * @return a new, non-{@code null} {@link Name}
207   *
208   * @exception IllegalArgumentException if {@code nameValue} is
209   * {@code null}
210   *
211   * @see Name#Name(Named, NameValue)
212   */
213  protected Name createName(final NameValue nameValue) {
214    return new Name(this, nameValue);
215  }
216
217  /**
218   * Removes the single {@link Name} stored under a {@link NameType}
219   * that is {@linkplain AbstractValued#equals(Object) equal} or
220   * identical to the supplied {@link NameType} and returns it.
221   *
222   * <p>This method may return {@code null} if there is no {@link
223   * Name} indexed under a {@link NameType} {@linkplain
224   * AbstractValued#equals(Object) equal} or identical to the supplied
225   * {@link NameType}.</p>
226   *
227   * @param nameType the {@link NameType} to use to find the {@link
228   * Name} in question; must not be {@code null}
229   *
230   * @return the {@link Name} that was removed, or {@code null}.  The
231   * {@link Name} that is returned, if non-{@code null}, will return
232   * {@code null} from its {@link Name#getNamed()} method.
233   *
234   * @exception IllegalArgumentException if {@code nameType} is {@code
235   * null}
236   */
237  public Name removeName(final NameType nameType) {
238    if (nameType == null) {
239      throw new IllegalArgumentException("nameType", new NullPointerException("nameType"));
240    }
241    Name returnValue = null;
242    if (this.names != null && !this.names.isEmpty()) {
243      returnValue = this.names.remove(nameType);
244    }
245    if (returnValue != null) {
246      returnValue.setNamed(null);
247    }
248    return returnValue;
249  }
250
251  /**
252   * Returns a non-{@code null} {@link Set} of {@link NameType}
253   * instances, each element of which can be used as an argument to
254   * the {@link #getName(NameType)} method.
255   *
256   * <p>This method never returns {@code null}.</p>
257   *
258   * @return a non-{@code null} {@link Set} of {@link NameType}s
259   */
260  public Set<NameType> getNameTypes() {
261    final Set<NameType> returnValue;
262    if (this.names == null || this.names.isEmpty()) {
263      returnValue = Collections.emptySet();
264    } else {
265      returnValue = Collections.unmodifiableSet(this.names.keySet());
266    }
267    return returnValue;
268  }
269
270  /**
271   * Returns a non-{@code null} {@link Collection} of {@link Name}s
272   * that this {@link AbstractNamed} is known by.
273   *
274   * <p>This method never returns {@code null}.</p>
275   *
276   * @return a non-{@code null} {@link Collection} of {@link Name}s
277   */
278  public Collection<Name> getNames() {
279    final Collection<Name> returnValue;
280    if (this.names == null || this.names.isEmpty()) {
281      returnValue = Collections.emptySet();
282    } else {
283      returnValue = Collections.unmodifiableCollection(this.names.values());
284    }
285    return returnValue;
286  }
