028package com.edugility.nomen;
029
030import java.beans.PropertyChangeEvent;
031import java.beans.PropertyChangeListener;
032
033import java.util.Iterator;
034import java.util.Map;
035import java.util.Map.Entry;
036
037/**
038 * A special-purpose {@link PropertyChangeListener} that ensures that
039 * if a {@link Name}'s ownership changes, then all entries within a
040 * {@link Map} that reference it are removed.
041 *
042 * <p>Consider the case of a {@link Map} that a {@link Named} uses to
043 * index its owned {@link Name}s under {@link NameType} keys.  A given
044 * {@link Name} might be indexed in this {@link Map} under two
045 * different {@link NameType}s.  If this {@link Name} changes its
046 * "{@code named}" (ownership) property from the {@link Named} in
047 * question to some other unrelated {@link Named}, then we need to
048 * ensure that all references to that {@link Name} in the {@link Map}
049 * are removed.</p>
050 *
051 * <p>The {@link NameSupport} class uses this class.  When any {@link
052 * Name} is added via the {@link NameSupport#putName(Map, NameType,
053 * Name)} method, a single instance of this class is {@linkplain
054 * Name#addPropertyChangeListener(String, PropertyChangeListener)
055 * added} as a {@link PropertyChangeListener} to the {@code named}
056 * property of that {@link Name}.</p>
057 *
058 * @author <a href="http://about.me/lairdnelson"
059 * target="_parent">Laird Nelson</a>
060 *
061 * @see PropertyChangeListener
062 *
063 * @see Name
064 *
065 * @see NameSupport
066 */
067public final class NameOwnershipMonitor implements PropertyChangeListener {
068
069
070  /*
071   * Instance fields.
072   */
073
074
075  /**
076   * The {@link Named} whose ownership of {@link Name}s is to be
077   * monitored.
078   *
079   * <p>This field will never be {@code null}.</p>
080   */
081  private final Named owner;
082
083  /**
084   * The set of {@link Entry}s representing the related owner's {@link
085   * Name}s.
086   *
087   * <p>This field will never be {@code null}.</p>
088   */
089  private final Iterable<? extends Entry<?, ? extends Name>> nameEntries;
090
091
092  /*
093   * Constructors.
094   */
095
096
097  /**
098   * Creates a new {@link NameOwnershipMonitor}.
099   *
100   * @param owner the {@link Named} whose ownership of {@link Name}s
101   * will be monitored; must not be {@code null}
102   *
103   * @param nameEntries an {@link Iterable} of {@link Entry} instances
104   * whose keys are usually {@link NameType}s and whose values are
105   * {@link Name}s; when a {@link Name} changes its "{@code named}"
106   * property then this {@link Iterable} will be combed to find any
107   * {@link Entry} instances that reference it and those instances
108   * will be removed
109   *
110   * @exception IllegalArgumentException if either parameter is {@code
111   * null}
112   *
113   * @see #propertyChange(PropertyChangeEvent)
114   */
115  public NameOwnershipMonitor(final Named owner, final Iterable<? extends Entry<?, ? extends Name>> nameEntries) {
116    super();
117    if (owner == null) {
118      throw new IllegalArgumentException("owner", new NullPointerException("owner"));
119    }
120    if (nameEntries == null) {
121      throw new IllegalArgumentException("nameEntries", new NullPointerException("nameEntries"));
122    }
123    this.owner = owner;
124    this.nameEntries = nameEntries;
125  }
126
127  
128  /*
129   * Instance methods.
130   */
131
132
133  /**
134   * Reacts to a change in the "{@code named}" property exposed by
135   * {@link Name} implementations by ensuring that any {@link Name}
136   * owned by the {@link Named} supplied to this class at construction
137   * time whose ownership is changing from that {@link Named} is
138   * removed from the {@link Iterable} supplied at construction time.
139   *
140   * <p>Basically, when a {@link Name} is put into a {@link Map}
141   * indexed by a {@link NameType} in some {@link Named} somewhere,
142   * then if that {@link Name}'s ownership changes "away" from that
143   * {@link Named}, that {@link Name} must be evicted from that {@link
144   * Map}, since it is no longer owned by that {@link Named}.  This
145   * method monitors a {@link Name} for that kind of ownership change,
146   * and performs the {@link Map} eviction appropriately.</p>
147   *
148   * @param event a {@link PropertyChangeEvent} describing a property
149   * change; must not be {@code null}
150   */
151  @Override
152  public final void propertyChange(final PropertyChangeEvent event) {
153    if (event != null && "named".equals(event.getPropertyName()) && this.owner == event.getOldValue()) {
154      final Object source = event.getSource();
155      if (source instanceof Name && this.owner != event.getNewValue() && this.nameEntries != null) {
156        final Iterator<? extends Entry<?, ? extends Name>> nameEntryIterator = this.nameEntries.iterator();
157        if (nameEntryIterator != null) {
158          while (nameEntryIterator.hasNext()) {
159            final Entry<?, ? extends Name> entry = nameEntryIterator.next();
160            if (entry != null) {
161              final Name n = entry.getValue();
162              if (n == source) {
163                // If we find a Name anywhere in the Entry set that is
164                // *the same* Name as the one whose ownership has
165                // changed, then remove that entry from the set, so
166                // there aren't any more NameTypes in the set that
167                // index it.
168                nameEntryIterator.remove();
169
170                // We remove ourselves from the name in question to
171                // avoid memory leaks.
172                n.removePropertyChangeListener("named", this);
173              }
174            }
175          }
176        }
177      }
178    }
179  }
