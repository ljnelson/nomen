032    /**
033     * An {@link AbstractValued} implementation that represents the value
034     * of a name of something.
035     *
036     * <p>In the nomenclature of the Nomen project, a {@linkplain Name
037     * <em>name</em>} is a {@linkplain NameValue <em>name value</em>} that
038     * is {@linkplain Name#getNamed() <em>owned</em>} by {@linkplain Named
039     * something that is <em>named</em>}.  Each concept has a
040     * corresponding class.</p>
041     *
042     * <p>A {@link NameValue} may be <em>atomic</em>&mdash;basically a
043     * glorified {@link String}, indivisible, not subject to any further
044     * interpretation&mdash;or not atomic&mdash;in which case it is
045     * interpreted as an <a href="http://mvel.codehaus.org/">MVEL</a> <a
046     * href="http://mvel.codehaus.org/MVEL+2.0+Templating+Guide">template</a>
047     * that will be {@linkplain Name#computeValue() interpolated} once the
048     * given {@link NameValue} is {@linkplain Name#getNameValue() owned}
049     * by a {@link Name}.  {@link NameValue}s are <em>not</em> atomic by
050     * default.</p>
051     *
052     * <p>A {@link NameValue} may have an {@linkplain
053     * #getWhitespaceReplacement() associated whitespace replacement
054     * <code>String</code>} that will be used to replace any sequence of
055     * one or more consecutive whitespace characters that occur after any
056     * {@linkplain Name#computeValue() interpolation} of a {@link
057     * NameValue} within the context of a given {@linkplain
058     * Name#getNamed() owner} when that {@link NameValue} is not
059     * {@linkplain #isAtomic() atomic}.</p>
060     *
061     * <p>A {@link NameValue}, once initialized with a {@linkplain
062     * #setValue(String) value} and an {@linkplain #setAtomic(boolean)
063     * atomicity}&mdash;whether via {@linkplain #NameValue(String,
064     * boolean, String) the constructor} or the {@link #setValue(String)}
065     * and {@link #setAtomic(boolean)} and {@link
066     * #setWhitespaceReplacement(String)} methods&mdash;can be treated as
067     * though it is immutable.  Subsequent attempts to call either the
068     * {@link #setValue(String)} or {@link #setAtomic(boolean)} or {@link
069     * #setWhitespaceReplacement(String)} methods will fail with {@link
070     * IllegalStateException}s.</p>
071     *
072     * <p>Two {@link NameValue}s are {@linkplain #equals(Object) equal} if
073     * their {@linkplain AbstractValued#getValue() values} are {@linkplain
074     * String#equals(Object) equal} and if they are both {@linkplain
075     * #isAtomic() atomic} and if their {@linkplain
076     * #getWhitespaceReplacement() whitespace replacement
077     * <code>String</code>}s are {@linkplain String#equals(Object)
078     * equal}.</p>
079     *
080     * <h3>Design Notes</h3>
081     *
082     * <p>This class is designed to be treated as immutable once fully
083     * {@linkplain #isInitialized() initialized}.  Consequently, certain
084     * methods, like {@link #setAtomic(boolean)} and {@link
085     * #setValue(String)} and {@link #setWhitespaceReplacement(String)},
086     * may only be called once with a given parameter value.</p>
087     *
088     * <p>Methods and fields that might otherwise be {@code final} are
089     * explicitly left non-{@code final} so that this class may be used as
090     * a JPA entity.</p>
091     *
092     * @author <a href="http://about.me/lairdnelson"
093     * target="_parent">Laird Nelson</a>
094     *
095     * @see Name
096     *
097     * @see AbstractValued
098     *
099     * @see Named
100     *
101     * @see #setValue(String)
102     *
103     * @see #setAtomic(boolean)
104     *
105     * @see #setWhitespaceReplacement(String)
106     *
107     * @see #isInitialized()
108     */
109    public class NameValue extends AbstractValued {

318      /**
319       * Returns the {@link String} that will be used instead of
320       * consecutive occurrences of one or more whitespace characters when
321       * this {@link NameValue}'s {@linkplain #getValue() value} is
322       * interpolated by a {@link Name}.
323       *
324       * <p>This method may return {@code null}, in which case no
325       * whitespace replacement will occur.</p>
326       *
327       * <p>This method always returns {@code null} if the value of the
328       * {@link #isAtomic()} method is {@code true}.  Since this method
329       * calls the {@link #isAtomic()} method, take care not to override
330       * that method to call this one, or an infinite loop will
331       * result.</p>
332       *
333       * @return a whitespace replacement {@link String}, or {@code null}
334       *
335       * @see #setWhitespaceReplacement(String)
336       *
337       * @see #isAtomic()
338       */
339      public String getWhitespaceReplacement() {
340        return this.isAtomic() ? null : this.whitespaceReplacement;
341      }

343      /**
344       * Sets the {@link String} that will be used instead of consecutive
345       * occurences of one or more whitespace characters when this {@link
346       * NameValue}'s {@linkplain #getValue() value} is interpolated by a
347       * {@link Name}.
348       *
349       * <p>If this {@link NameValue} {@linkplain #isAtomic() is atomic},
350       * then the only permissible value for the {@code
351       * whitespaceReplacement} parameter is {@code null}.</p>
352       *
353       * @param whitespaceReplacement the replacement {@link String}; if
354       * {@code null} then no whitespace replacement will occur
355       *
356       * @exception IllegalArgumentException if this {@link NameValue}
357       * {@linkplain #isAtomic() is atomic} and {@code
358       * whitespaceReplacement} is not {@code null}
359       *
360       * @exception IllegalStateException if this method is called more
361       * than once with different values for the {@code
362       * whitespaceReplacement} parameter
363       *
364       * @see Name#getValue()
365       */
366      public void setWhitespaceReplacement(final String whitespaceReplacement) {
367        if (this.whitespaceReplacementSet) {
368          if (whitespaceReplacement == null) {
369            if (this.getWhitespaceReplacement() != null) {
370              throw new IllegalStateException("setWhitespaceReplacement() cannot be called more than once with different parameter values");
371            }
372          } else if (!whitespaceReplacement.equals(this.getWhitespaceReplacement())) {
373            throw new IllegalStateException("setWhitespaceReplacement() cannot be called more than once with different parameter values");
374          }
375        } else {
376          if (this.isAtomic() && whitespaceReplacement != null) {
377            throw new IllegalArgumentException("whitespaceReplacement", new IllegalStateException("isAtomic()"));
378          }
379          this.whitespaceReplacement = whitespaceReplacement;
380          this.whitespaceReplacementSet = true;
381        }
382      }

405       * Returns {@code true} if this {@link NameValue} is <em>atomic</em>
406       * &mdash;if it is a simple textual value and therefore not a <a

414       *

416       * textual value; {@code false} otherwise
417       *
418       * @see #getWhitespaceReplacement()
419       */

421        return this.atomic != null && this.atomic.booleanValue();
422      }

424      /**
425       * Sets whether this {@link NameValue} is <em>atomic</em> or
426       * not&mdash;whether it is a simple textual value or is a <a
427       * href="http://mvel.codehaus.org/MVEL+2.0+Templating+Guide">template</a>
428       * that must be interpolated.
429       *
430       * <p>This method may only be called once to set the initial value
431       * for this property.  If it is called again, an {@link
432       * IllegalStateException} will be thrown.</p>
433       *
434       * @param atomic whether this {@link NameValue} is <em>atomic</em>
435       * or not&mdash;whether it is a simple textual value or is a <a
436       * href="http://mvel.codehaus.org/MVEL+2.0+Templating+Guide">template</a>
437       * that must be interpolated
438       *
439       * @exception IllegalStateException if this method is called more
440       * than once
441       */
442      public void setAtomic(final boolean atomic) {
443        if (this.atomic == null) {
444          this.atomic = Boolean.valueOf(atomic);
445        } else if (this.atomic.booleanValue() != atomic) {
446          throw new IllegalStateException("setAtomic() cannot be called more than once with different parameter values");
447        }
448      }

450      /**
451       * Returns a hashcode for this {@link NameValue} based off the
452       * return value of this {@link NameValue}'s {@link #isAtomic()}
453       * method, the return value of this {@link NameValue}'s {@link
454       * #getWhitespaceReplacement()} method and its {@linkplain
455       * AbstractValued#hashCode() superclass' hashcode}.
456       *
457       * @return a hashcode for this {@link NameValue}
458       *
459       * @see #equals(Object)
460       */
461      @Override
462      public int hashCode() {
463        int result = 17;
464        
465        int c = Boolean.valueOf(this.isAtomic()).hashCode();
466        result = result * 37 + c;
467        
468        final String whitespaceReplacement = this.getWhitespaceReplacement();
469        if (whitespaceReplacement == null) {
470          c = 0;
471        } else {
472          c = whitespaceReplacement.hashCode();
473        }
474        result = result * 37 + c;
475    
476        c = super.hashCode();
477        result = result * 37 + c;
478        
479        return result;
480      }

482      /**
483       * Returns {@code true} if the supplied {@link Object} is an
484       * instance of {@link NameValue} and returns the same value from its
485       * {@link #isAtomic()} method as is returned from this {@link
486       * NameValue}'s {@link #isAtomic()} method and returns a value from
487       * its {@link #getWhitespaceReplacement()} method that is
488       * {@linkplain String#equals(Object) equal to} the value returned by
489       * this {@link NameValue}'s {@link #getWhitespaceReplacement()}
490       * method and returns a value from its {@link
491       * AbstractValued#getValue()} method that is {@linkplain
492       * String#equals(Object) equal to} the value returned by this {@link
493       * NameValue}'s {@link AbstractValued#getValue()} method.
494       *
495       * @param other the {@link Object} to compare this {@link NameValue}
496       * against for equality; may be {@code null}
497       *
498       * @return {@code true} if the supplied {@link Object} is equal to
499       * this {@link NameValue}; {@code false} otherwise
500       *
501       * @see #hashCode()
502       *
503       * @see AbstractValued#equals(Object)
504       */
505      @Override
506      public boolean equals(final Object other) {
507        if (other == this) {
508          return true;
509        } else if (other instanceof NameValue) {
510          final NameValue him = (NameValue)other;
511    
512          if (this.isAtomic()) {
513            if (!him.isAtomic()) {
514              return false;
515            }
516          }
517    
518          final String whitespaceReplacement = this.getWhitespaceReplacement();
519          if (whitespaceReplacement == null) {
520            if (him.getWhitespaceReplacement() != null) {
521              return false;
522            }
523          } else if (!whitespaceReplacement.equals(him.getWhitespaceReplacement())) {
524            return false;
525          }
526          
527          return super.equals(other);
528        } else {
529          return false;
530        }
531      }
