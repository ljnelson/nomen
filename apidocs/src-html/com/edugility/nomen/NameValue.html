028    package com.edugility.nomen;
029    
030    import java.io.Serializable;
031    
032    /**
033     * An {@link AbstractValued} implementation that represents the value
034     * of a name of something.
035     *
036     * <p>In the nomenclature of the Nomen project, a {@linkplain Name
037     * <em>name</em>} is a {@linkplain NameValue <em>name value</em>} that
038     * is {@linkplain Name#getNamed() <em>owned</em>} by {@linkplain Named
039     * something that is <em>named</em>}.  Each concept has a
040     * corresponding class.</p>
041     *
042     * <p>A {@link NameValue} may be <em>atomic</em>&mdash;basically a
043     * glorified {@link String}, indivisible, not subject to any further
044     * interpretation&mdash;or not atomic&mdash;in which case it is
045     * interpreted as an <a href="http://mvel.codehaus.org/">MVEL</a>
046     * template that will be {@linkplain Name#computeValue() interpolated}
047     * once the given {@link NameValue} is {@linkplain Name#getNameValue()
048     * owned} by a {@link Name}.  {@link NameValue}s are <em>not</em>
049     * atomic by default.</p>
050     *
051     * <p>A {@link NameValue} may have an {@linkplain
052     * #getWhitespaceReplacement() associated whitespace replacement
053     * <code>String</code>} that will be used to replace any sequence of
054     * one or more consecutive whitespace characters that occur after any
055     * {@linkplain Name#computeValue() interpolation} of a {@link
056     * NameValue} within the context of a given {@linkplain
057     * Name#getNamed() owner} when that {@link NameValue} is not
058     * {@linkplain #isAtomic() atomic}.</p>
059     *
060     * <p>A {@link NameValue}, once initialized with a {@linkplain
061     * #setValue(String) value} and an {@linkplain #setAtomic(boolean)
062     * atomicity}&mdash;whether via {@linkplain #NameValue(String,
063     * boolean, String) the constructor} or the {@link #setValue(String)}
064     * and {@link #setAtomic(boolean)} and {@link
065     * #setWhitespaceReplacement(String)} methods&mdash;can be treated as
066     * though it is immutable.  Subsequent attempts to call either the
067     * {@link #setValue(String)} or {@link #setAtomic(boolean)} or {@link
068     * #setWhitespaceReplacement(String)} methods will fail with {@link
069     * IllegalStateException}s.</p>
070     *
071     * <p>Two {@link NameValue}s are {@linkplain #equals(Object) equal} if
072     * their {@linkplain AbstractValued#getValue() values} are {@linkplain
073     * String#equals(Object) equal} and if they are both {@linkplain
074     * #isAtomic() atomic} and if their {@linkplain
075     * #getWhitespaceReplacement() whitespace replacement
076     * <code>String</code>}s are {@linkplain String#equals(Object)
077     * equal}.</p>
078     *
079     * <h3>Design Notes</h3>
080     *
081     * <p>Methods and fields that might otherwise be {@code final} are
082     * explicitly left non-{@code final} so that this class may be used as
083     * a JPA entity.  Certain methods, like {@link #setAtomic(boolean)}
084     * and {@link #setValue(String)}, may only be called once.</p>
085     *
086     * @author <a href="http://about.me/lairdnelson"
087     * target="_parent">Laird Nelson</a>
088     *
089     * @see Name
090     *
091     * @see AbstractValued
092     *
093     * @see Named
094     */
095    public class NameValue extends AbstractValued {

117      /**
118       * Whether this {@link NameValue} is considered to be
119       * <em>atomic</em>&mdash;i.e. not a template in need of further
120       * interpolation, but just a simple textual value.
121       *
122       * <p>This field may be {@code null}.</p>
123       *
124       * <h4>Design Notes</h4>
125       *
126       * <p>This field is a {@link Boolean} and not a {@code boolean} so
127       * that its initial setting via the {@link #setAtomic(boolean)}
128       * method can be tracked.  It is not {@code final} so that this
129       * class may be used as a JPA entity.</p>
130       *
131       * @see #isAtomic()
132       *
133       * @see #setAtomic(boolean)
134       */
135      private Boolean atomic;

137      /**
138       * A {@link String} that will replace one or more consecutive

142       * <p>This field may be {@code null} in which case no whitespace
143       * replacement will occur.</p>
144       *
145       * @see #getWhitespaceReplacement()
146       *
147       * @see #setWhitespaceReplacement(String)
148       */
149      private String whitespaceReplacement;
150    
151      /**
152       * Indicates whether the {@link #whitespaceReplacement} field has
153       * been set via the {@link #setWhitespaceReplacement(String)} method
154       * or not.
155       *
156       * @see #setWhitespaceReplacement(String)
157       */
158      private boolean whitespaceReplacementSet;

166      /**
167       * Creates a new {@link NameValue} in an incomplete state; to fully
168       * initialize the new {@link NameValue} callers must invoke the

170       * #setAtomic(boolean)} method.

172       * <h4>Design Notes</h4>

174       * <p>This constructor exists for JPA compatibility.  Please

176       * constructor instead which fully initializes a {@link NameValue}
177       * instance as it creates it.</p>

186       * Creates a new {@link NameValue} that is not {@linkplain

192       *
193       * @param value the {@linkplain #setValue(String) value} for this
194       * {@link NameValue}; must not be {@code null}

199       * @see #NameValue(String, boolean, String)
200       */

202        this(value, false, " ");
203      }

206       * Creates a new {@link NameValue}.

208       * @param value the {@linkplain #setValue(String) value} for this
209       * {@link NameValue}; must not be {@code null}

211       * @param atomic whether or not this new {@link NameValue} will be

221      /**
222       * Creates a new {@link NameValue}.
223       *
224       * <p>This constructor calls the {@link #NameValue(String, boolean,
225       * String)} constructor with {@code false} as the value for its
226       * second parameter and a space (" ") as the value for its third
227       * parameter.</p>
228       *
229       * @param value the {@linkplain #setValue(String) value} for this
230       * {@link NameValue}; must not be {@code null}
231       *
232       * @param atomic the {@linkplain #setAtomic(boolean) atomicity} for
233       * this {@link NameValue}
234       *
235       * @param whitespaceReplacement the {@link String} to be used for
236       * whitespace replacement; may be {@code null} in which case no
237       * whitespace replacement will be performed.  Normally, a single
238       * space (" ") is a good value for this parameter.
239       *
240       * @exception IllegalArgumentException if {@code value} is {@code
241       * null}
242       */
243      public NameValue(final String value, final boolean atomic, final String whitespaceReplacement) {
244        super(value);
245        this.setAtomic(atomic);
246        this.setWhitespaceReplacement(whitespaceReplacement);
247      }

255      /**
256       * Returns the {@link String} that will be used instead of
257       * consecutive occurrences of one or more whitespace characters when
258       * this {@link NameValue}'s {@linkplain #getValue() value} is
259       * interpolated by a {@link Name}.
260       *
261       * <p>This method may return {@code null}, in which case no
262       * whitespace replacement will occur.</p>
263       *
264       * @return a whitespace replacement {@link String}, or {@code null}
265       *
266       * @see #setWhitespaceReplacement(String)
267       */
268      public String getWhitespaceReplacement() {
269        return this.whitespaceReplacement;
270      }
271    
272      /**
273       * Sets the {@link String} that will be used instead of consecutive
274       * occurences of one or more whitespace characters when this {@link
275       * NameValue}'s {@linkplain #getValue() value} is interpolated by a
276       * {@link Name}.
277       *
278       * @param whitespaceReplacement the replacement {@link String}; if
279       * {@code null} then no whitespace replacement will occur
280       *
281       * @see Name#getValue()
282       */
283      public void setWhitespaceReplacement(final String whitespaceReplacement) {
284        if (this.whitespaceReplacementSet) {
285          if (whitespaceReplacement == null) {
286            if (this.getWhitespaceReplacement() != null) {
287              throw new IllegalStateException();
288            }
289          } else if (!whitespaceReplacement.equals(this.getWhitespaceReplacement())) {
290            throw new IllegalStateException();
291          }
292        } else {
293          this.whitespaceReplacement = whitespaceReplacement;
294          this.whitespaceReplacementSet = true;
295        }
296      }
297      
298      /**
299       * Returns {@code true} if this {@link NameValue} is fully
300       * initialized and hence immutable.
301       *
302       * @return {@code true} if this {@link NameValue} is fully
303       * initialized
304       *
305       * @see #setValue(String)
306       *
307       * @see #setAtomic(boolean)
308       */
309      public boolean isInitialized() {
310        return this.atomic != null && this.getValue() != null && this.whitespaceReplacementSet;
311      }
312    
313      /**
314       * Returns {@code true} if this {@link NameValue} is <em>atomic</em>
315       * &mdash;if it is a simple textual value and therefore not a
316       * template that must be interpolated.
317       *
318       * @return {@code true} if this {@link NameValue} is an indivisible
319       * textual value; {@code false} otherwise
320       */
321      public boolean isAtomic() {
322        return this.atomic != null && this.atomic.booleanValue();
323      }
324    
325      /**
326       * Sets whether this {@link NameValue} is <em>atomic</em> or
327       * not&mdash;whether it is a simple textual value or is a template
328       * that must be interpolated.
329       *
330       * <p>This method may only be called once to set the initial value
331       * for this property.  If it is called again, an {@link
332       * IllegalStateException} will be thrown.</p>
333       *
334       * @param atomic whether this {@link NameValue} is <em>atomic</em> or
335       * not&mdash;whether it is a simple textual value or is a template
336       * that must be interpolated
337       *
338       * @exception IllegalStateException if this method is called more
339       * than once
340       */
341      public void setAtomic(final boolean atomic) {
342        if (this.atomic == null) {
343          this.atomic = Boolean.valueOf(atomic);
344        } else if (this.atomic.booleanValue() != atomic) {
345          throw new IllegalStateException();
346        }
347      }
348    
349      /**
350       * Returns a hashcode for this {@link NameValue} based off the
351       * return value of this {@link NameValue}'s {@link #isAtomic()}
352       * method, the return value of this {@link NameValue}'s {@link
353       * #getWhitespaceReplacement()} method and its {@linkplain
354       * AbstractValued#hashCode() superclass' hashcode}.
355       *
356       * @return a hashcode for this {@link NameValue}
357       */
358      @Override
359      public int hashCode() {
360        int result = 17;
361        
362        int c = Boolean.valueOf(this.isAtomic()).hashCode();
363        result = result * 37 + c;
364        
365        final String whitespaceReplacement = this.getWhitespaceReplacement();
366        if (whitespaceReplacement == null) {
367          c = 0;
368        } else {
369          c = whitespaceReplacement.hashCode();
370        }
371        result = result * 37 + c;
372    
373        c = super.hashCode();
374        result = result * 37 + c;
375        
376        return result;
377      }
378    
379      /**
380       * Returns {@code true} if the supplied {@link Object} is an
381       * instance of {@link NameValue} and returns the same value from its
382       * {@link #isAtomic()} method as is returned from this {@link
383       * NameValue}'s {@link #isAtomic()} method and returns a value from
384       * its {@link #getWhitespaceReplacement()} method that is
385       * {@linkplain String#equals(Object) equal to} the value returned by
386       * this {@link NameValue}'s {@link #getWhitespaceReplacement()}
387       * method and returns a value from its {@link
388       * AbstractValued#getValue()} method that is {@linkplain
389       * String#equals(Object) equal to} the value returned by this {@link
390       * NameValue}'s {@link AbstractValued#getValue()} method.
391       *
392       * @param other the {@link Object} to compare this {@link NameValue}
393       * against for equality; may be {@code null}
394       *
395       * @return {@code true} if the supplied {@link Object} is equal to
396       * this {@link NameValue}; {@code false} otherwise
397       *
398       * @see AbstractValued#equals(Object)
399       */
400      @Override
401      public boolean equals(final Object other) {
402        if (other == this) {
403          return true;
404        } else if (other instanceof NameValue) {
405          final NameValue him = (NameValue)other;
406    
407          if (this.isAtomic()) {
408            if (!him.isAtomic()) {
409              return false;
410            }
411          }
412    
413          final String whitespaceReplacement = this.getWhitespaceReplacement();
414          if (whitespaceReplacement == null) {
415            if (him.getWhitespaceReplacement() != null) {
416              return false;
417            }
418          } else if (!whitespaceReplacement.equals(him.getWhitespaceReplacement())) {
419            return false;
420          }
421          
422          return super.equals(other);
423        } else {
424          return false;
425        }
426      }

523      /**
524       * Returns a non-{@code null} {@link NameValue} whose {@link
525       * #getValue()} method is guaranteed to return a non-{@code null}
526       * {@link String} that is {@linkplain String#equals(Object) equal
527       * to} the supplied {@code value} and whose {@link #isAtomic()}
528       * method will return a value equal to the supplied {@code atomic}
529       * parameter.
530       *
531       * <p>This method never returns {@code null}.</p>
532       *
533       * @param value the value for which a {@link NameValue} should be
534       * returned; must not be {@code null}
535       *
536       * @param atomic whether the {@link NameValue} returned {@linkplain
537       * #isAtomic() is atomic}
538       *
539       * @param whitespaceReplacement the {@link String} that will be used
540       * instead of one or more consecutive occurrences of whitespace
541       * characters in the {@linkplain Name#getValue() value of this
542       * <code>NameValue</code> when interpolated by an owning
543       * <code>Name</code>}; may be {@code null} in which case no
544       * whitespace replacement will occur

546       * @return a non-{@code null} {@link NameValue} whose {@link
547       * #getValue()} method will return a {@link String} equal to the
548       * supplied {@link String} and whose {@link #isAtomic()} method will
549       * return a value equal to the supplied {@code atomic} parameter
550       *
551       * @exception IllegalArgumentException if {@code value} is {@code
552       * null}
553       */
554      public static final NameValue valueOf(final String value, final boolean atomic, final String whitespaceReplacement) {
555        if (value == null) {
556          throw new IllegalArgumentException("value", new NullPointerException("value"));
557        }
558        return new NameValue(value, atomic, whitespaceReplacement);
559      }
