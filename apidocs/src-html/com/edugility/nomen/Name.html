051/**
052 * A {@link Serializable} {@link AbstractValued} implementation that
053 * is a combination of a {@link NameValue} and the {@link Named} that
054 * it is currently assigned to.  A {@link Name} joins a {@link
055 * NameValue} with a {@link Named} so that {@link NameValue}s can be
056 * shared among several {@link Named}s.
057 *
058 * <p>A {@link Name} may be a name for a party or agent or less
059 * tangible concept.  There is no guarantee that a {@link Name} serves
060 * as any kind of unique identifier.</p>
061 *
062 * <p>In normal usage, a {@link Name} is set up and then not further
063 * altered.  However, it can be {@linkplain #setNamed(Named)
064 * reassigned} and have its {@linkplain #setNameValue(NameValue) value
065 * changed}.  Both its {@code named} and {@code nameValue} properties
066 * are bound and thus fire {@link PropertyChangeEvent}s when
067 * altered.</p>
068 *
069 * <p>Two {@link Name}s are considered {@linkplain #equals(Object)
070 * equal} if their {@linkplain #getValue() values} are equal.  Note in
071 * particular that for {@linkplain #equals(Object) equality purposes}
072 * <strong>a {@link Name}'s {@linkplain #getNamed() owner} is deliberately not
073 * also considered.</strong></p>
074 *
075 * <p>The {@linkplain #getValue() value} of a non-{@linkplain
076 * NameValue#isAtomic() atomic} {@link Name} may change as its
077 * {@linkplain #getNamed() associated owner}'s set of names changes.
078 * Bear this in mind if you are using {@link Name}s as keys in a
079 * {@link Map} (generally a bad idea).</p>
080 *
081 * @author <a href="http://about.me/lairdnelson"
082 * target="_parent">Laird Nelson</a>
083 *
084 * @see #getValue()
085 *
086 * @see #computeValue()
087 *
088 * @see NameValue
089 *
090 * @see Named
091 *
092 * @see Valued
093 *
094 */

328  /**
329   * Installs the {@link NameValue} that this {@link Name}
330   * fundamentally has as its value.
331   *
332   * @param nameValue the {@link NameValue} to install; must not be
333   * {@code null}
334   *
335   * @exception IllegalArgumentException if {@code nameValue} is
336   * {@code null} or {@linkplain NameValue#isInitialized() not
337   * initialized}
338   *
339   * @exception IllegalStateException if the supplied {@link
340   * NameValue} is {@linkplain NameValue#isAtomic() not atomic} and
341   * there was a problem interpolating the template it logically
342   * represents; examine the {@link Throwable#getCause() root cause}
343   * for details
344   *
345   * @see #getNameValue()
346   *
347   * @see NameValue#isAtomic()
348   *
349   * @see NameValue#setAtomic(boolean)
350   */
351  public void setNameValue(final NameValue nameValue) {
352    if (nameValue == null) {
353      throw new IllegalArgumentException("nameValue", new NullPointerException("nameValue"));
354    } else if (!nameValue.isInitialized()) {
355      throw new IllegalArgumentException("!nameValue.isInitialized()");
356    }
357    final NameValue old = this.getNameValue();
358    if (!nameValue.equals(old)) {
359      this.nameValue = nameValue;
360      this.compiledTemplate = null;
361      this.installTemplate();
362      this.firePropertyChange("nameValue", old, this.getNameValue());
363    }
364  }
365
366  /**
367   * If the {@link #compiledTemplate} field is {@code null} and the
368   * {@linkplain #getNameValue() affiliated <code>NameValue</code>} is
369   * non-{@code null}, {@linkplain NameValue#isAtomic() is not atomic}
370   * and its {@linkplain NameValue#getValue() value} is non-{@code
371   * null}, {@linkplain TemplateCompiler#compileTemplate(String)
372   * compiles} that value for quick {@linkplain
373   * TemplateRuntime#execute(CompiledTemplate, Object,
374   * VariableResolverFactory) execution} later by the {@link
375   * #getValue()} method.
376   *
377   * <h3>Design Notes</h3>
378   *
379   * <p>This method is not {@code final} only so this class can be
380   * used as a JPA entity.<p>
381   *
382   * @exception IllegalStateException if a template compilation error
383   * occurs
384   *
385   * @see #getNameValue()
386   *
387   * @see NameValue#getValue()
388   *
389   * @see #getValue()
390   */
391  private void installTemplate() {
392    if (this.compiledTemplate == null) {
393      final NameValue nv = this.getNameValue();
394      if (nv != null && !nv.isAtomic()) {
395        final String template = nv.getValue();
396        if (template != null) {
397          try {
398            this.compiledTemplate = TemplateCompiler.compileTemplate(template);
399          } catch (final CompileException wrapMe) {
400            throw new IllegalStateException(wrapMe);
401          }
402          assert this.compiledTemplate != null;
403          this.firePropertyChange("compiledTemplate", null, this.compiledTemplate);
404        }          
405      }
406    }    
407  }

462  /**
463   * Computes and returns the most up-to-date value possible for this
464   * {@link Name}.
465   *
466   * <p>This method never returns {@code null}.</p>
467   *
468   * <p>This method is not declared {@code final} only so that this
469   * class may be used as a JPA entity.</p>
470   *
471   * @return a non-{@code null} {@link String} representing this
472   * {@link Name}'s value
473   *
474   * @exception IllegalStateException if there was a problem compiling
475   * a template
476   */
477  protected String computeValue() {
478    final String returnValue;
479    final NameValue nv = this.getNameValue();
480    if (nv == null) {
481      returnValue = "";
482    } else if (nv.isAtomic() || this.compiledTemplate == null) {
483      final Object rawValue = nv.getValue();
484      if (rawValue == null) {
485        returnValue = "";
486      } else {
487        returnValue = this.toString(rawValue);
488      }
489    } else {
490      Object rawValue = null;
491      try {
492        rawValue = TemplateRuntime.execute(this.compiledTemplate, this.getNamed(), this.nameResolverFactory);
493      } catch (final IllegalStateException throwMe) {
494        throw throwMe;
495      } catch (final RuntimeException wrapMe) {          
496        throw new IllegalStateException(wrapMe);
497      }
498      final String rawStringValue = this.toString(rawValue);
499      if (rawStringValue == null || rawStringValue.isEmpty()) {
500        returnValue = "";
501      } else {
502        final String whitespaceReplacement = nv.getWhitespaceReplacement();
503        if (whitespaceReplacement != null) {
504          returnValue = whitespacePattern.matcher(rawStringValue).replaceAll(whitespaceReplacement);
505        } else {
506          returnValue = rawStringValue;
507        }
508      }
509    }
510    assert returnValue != null;
511    return returnValue;
512  }
