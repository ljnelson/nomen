050    /**
051     * A {@link Serializable} {@link AbstractValued} implementation that
052     * is a combination of a {@link NameValue} and the {@link Named} that
053     * it is currently assigned to.
054     *
055     * <p>A {@link Name} may be a name for a party or agent or less
056     * tangible concept.  There is no guarantee that a {@link Name} serves
057     * as any kind of unique identifier.</p>
058     *
059     * <p>Two {@link Name}s are considered {@linkplain #equals(Object)
060     * equal} if their {@linkplain #getValue() values} are equal.  Note in
061     * particular that for {@linkplain #equals(Object) equality purposes}
062     * a {@link Name}'s {@linkplain #getNamed() owner} is deliberately not
063     * also considered.</p>
064     *
065     * <p>The {@linkplain #getValue() value} of a {@link Name} may change,
066     * since the {@linkplain NameValue#getValue() value of its associated
067     * <code>NameValue</code>} may change.  Bear this in mind if you are
068     * using {@link Name}s as keys in a {@link Map}.  In normal usage,
069     * however, a {@link Name} is set up and then not further altered.</p>
070     *
071     * @author <a href="http://about.me/lairdnelson"
072     * target="_parent">Laird Nelson</a>
073     *
074     * @see #getValue()
075     *
076     * @see #computeValue()
077     *
078     * @see NameValue
079     *
080     * @see Named
081     *
082     * @see Valued
083     */

293      /**
294       * Installs the {@link NameValue} that this {@link Name}
295       * fundamentally has as its value.
296       *
297       * @param nameValue the {@link NameValue} to install; must not be
298       * {@code null}
299       *
300       * @exception IllegalArgumentException if {@code nameValue} is
301       * {@code null} or {@linkplain NameValue#isInitialized() not
302       * initialized}
303       *
304       * @exception IllegalStateException if the supplied {@link
305       * NameValue} is {@linkplain NameValue#isAtomic() not atomic} and
306       * there was a problem interpolating the template it logically
307       * represents; examine the {@link Throwable#getCause() root cause}
308       * for details
309       *
310       * @see #getNameValue()
311       *
312       * @see NameValue#isAtomic()
313       *
314       * @see NameValue#setAtomic(boolean)
315       */
316      public void setNameValue(final NameValue nameValue) {
317        if (nameValue == null) {
318          throw new IllegalArgumentException("nameValue", new NullPointerException("nameValue"));
319        } else if (!nameValue.isInitialized()) {
320          throw new IllegalArgumentException("!nameValue.isInitialized()");
321        }
322        final NameValue old = this.getNameValue();
323        if (!nameValue.equals(old)) {
324          this.nameValue = nameValue;
325          this.compiledTemplate = null;
326          this.installTemplate();
327          this.firePropertyChange("nameValue", old, this.getNameValue());
328        }
329      }

331      /**
332       * If the {@link #compiledTemplate} field is {@code null} and the
333       * {@linkplain #getNameValue() affiliated <code>NameValue</code>} is
334       * non-{@code null}, {@linkplain NameValue#isAtomic() is not atomic}
335       * and its {@linkplain NameValue#getValue() value} is non-{@code
336       * null}, {@linkplain TemplateCompiler#compileTemplate(String)
337       * compiles} that value for quick {@linkplain
338       * TemplateRuntime#execute(CompiledTemplate, Object,
339       * VariableResolverFactory) execution} later by the {@link
340       * #getValue()} method.
341       *
342       * <h4>Design Notes</h4>
343       *
344       * <p>This method is not {@code final} only so this class can be
345       * used as a JPA entity.<p>
346       *
347       * @exception IllegalStateException if a template compilation error
348       * occurs
349       *
350       * @see #getNameValue()
351       *
352       * @see NameValue#getValue()
353       *
354       * @see #getValue()
355       */
356      private void installTemplate() {
357        if (this.compiledTemplate == null) {
358          final NameValue nv = this.getNameValue();
359          if (nv != null && !nv.isAtomic()) {
360            final String template = nv.getValue();
361            if (template != null) {
362              try {
363                this.compiledTemplate = TemplateCompiler.compileTemplate(template);
364              } catch (final CompileException wrapMe) {
365                throw new IllegalStateException(wrapMe);
366              }
367              assert this.compiledTemplate != null;
368              this.firePropertyChange("compiledTemplate", null, this.compiledTemplate);
369            }          
370          }
371        }    
372      }

427      /**
428       * Computes and returns the most up-to-date value possible for this
429       * {@link Name}.
430       *
431       * <p>This method never returns {@code null}.</p>
432       *
433       * <p>This method is not declared {@code final} only so that this
434       * class may be used as a JPA entity.</p>
435       *
436       * @return a non-{@code null} {@link String} representing this
437       * {@link Name}'s value
438       *
439       * @exception IllegalStateException if there was a problem compiling
440       * a template
441       */
442      protected String computeValue() {
443        final String returnValue;
444        final NameValue nv = this.getNameValue();
445        if (nv == null) {
446          returnValue = "";
447        } else if (nv.isAtomic() || this.compiledTemplate == null) {
448          final String rawValue = nv.getValue();
449          if (rawValue == null) {
450            returnValue = "";
451          } else {
452            returnValue = rawValue;
453          }
454        } else {
455          Object rawValue = null;
456          try {
457            rawValue = TemplateRuntime.execute(this.compiledTemplate, this.getNamed(), this.nameResolverFactory);
458          } catch (final IllegalStateException throwMe) {
459            throw throwMe;
460          } catch (final RuntimeException wrapMe) {          
461            throw new IllegalStateException(wrapMe);
462          }
463          final String rawStringValue = this.toString(rawValue);
464          if (rawStringValue == null || rawStringValue.isEmpty()) {
465            returnValue = "";
466          } else {
467            final String whitespaceReplacement = nv.getWhitespaceReplacement();
468            if (whitespaceReplacement != null) {
469              returnValue = whitespacePattern.matcher(rawStringValue).replaceAll(whitespaceReplacement);
470            } else {
471              returnValue = rawStringValue;
472            }
473          }
474        }
475        assert returnValue != null;
476        return returnValue;
477      }

809       * A convenience method that creates and installs a new
810       * non-{@linkplain NameValue#isAtomic() atomic} {@link Name} into
811       * the supplied {@link AbstractNamed}.
812       *
813       * <h4>Design Notes</h4>
814       *
815       * <p>This method is not declared {@code final} only so that this
816       * class may be used as a JPA entity.  The JPA specification
817       * requires that no method on a potential entity may be {@code
818       * final}.</p>
819       *
820       * @param named the {@link AbstractNamed} that will own the new
821       * {@link Name}; must not be {@code null}
822       * 
823       * @param nameTypeValue the identifying information for the relevant
824       * {@link NameType} under which a new {@link Name} will be indexed; must not be {@code null}
825       *
826       * @param nameValue the identifying information for the relevant
827       * {@link NameValue}; must not be {@code null}
828       *
829       * @param atomic whether the {@code nameValue} parameter is
830       * notionally {@linkplain NameValue#isAtomic() atomic}
831       *
832       * @param whitespaceReplacement the {@link String} to use in
833       * replacing {@linkplain Character#isWhitespace(char) whitespace
834       * characters}; may be {@code null} in which case no whitespace
835       * replacement will occur.  A good reasonable default is a single
836       * space (" ").
837       *
838       * @return a non-{@code null} {@link Name} {@linkplain #getNamed()
839       * owned} by the supplied {@link AbstractNamed} under an appropriate
840       * {@link NameType}
841       *
842       * @exception IllegalArgumentException if either {@code named},
843       * {@code nameTypeValue} or {@code nameValue} is {@code null}.
844       */
845      public static Name name(final AbstractNamed named, final String nameTypeValue, final String nameValue, final boolean atomic, final String whitespaceReplacement) {
846        if (named == null) {
847          throw new IllegalArgumentException("named", new NullPointerException("named"));
848        }
849        final NameType nameType = NameType.valueOf(nameTypeValue);
850        assert nameType != null;    
851        named.putName(nameType, NameValue.valueOf(nameValue, atomic, whitespaceReplacement));
852        return named.getName(nameType);
853      }
