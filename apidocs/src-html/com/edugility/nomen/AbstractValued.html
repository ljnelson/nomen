028    package com.edugility.nomen;
029    
030    import java.io.Serializable;
031    
032    /**
033     * A {@link Serializable} {@link Valued} implementation that correctly
034     * implements the {@link #equals(Object)} and {@link #hashCode()}
035     * methods.
036     *
037     * @author <a href="http://about.me/lairdnelson"
038     * target="_parent">Laird Nelson</a>
039     *
040     * @see #getValue()
041     */
042    public abstract class AbstractValued implements Serializable, Valued {
043    
044    
045      /*
046       * Static fields.
047       */
048    
049    
050      /**
051       * The version of this class for {@linkplain Serializable
052       * serialization purposes}.
053       *
054       * @see Serializable
055       */
056      private static final long serialVersionUID = 1L;
057    
058    
059      /*
060       * Instance fields.
061       */
062    
063    
064      /**
065       * The value of this {@link AbstractValued}.
066       *
067       * <p>This field may be {@code null}.</p>
068       *
069       * @see #getValue()
070       *
071       * @see #setValue(String)
072       */
073      private String value;
074    
075    
076      /*
077       * Constructors.
078       */
079    
080    
081      /**
082       * Creates a new {@link AbstractValued}.
083       */
084      protected AbstractValued() {
085        super();
086      }
087    
088      /**
089       * Creates a new {@link AbstractValued} with the supplied {@code
090       * value}.
091       *
092       * @param value the new value; must not be {@code null}
093       *
094       * @exception IllegalArgumentException if {@code value} is {@code
095       * null}
096       *
097       * @see #setValue(String)
098       */
099      protected AbstractValued(final String value) {
100        this();
101        this.setValue(value);
102      }
103    
104    
105      /*
106       * Instance methods.
107       */
108    
109    
110      /**
111       * Returns the value of this {@link AbstractValued}.
112       *
113       * <p>This method may return {@code null}.</p>
114       *
115       * @return the value of this {@link AbstractValued}, or {@code
116       * null}
117       */
118      @Override
119      public String getValue() {
120        return this.value;
121      }
122    
123      /**
124       * Sets the value of this {@link AbstractValued}.
125       *
126       * <p>The default implementation of this method may only be called
127       * once.  Subsequent calls will throw an {@link
128       * IllegalStateException}.  Subclasses are free to override this
129       * restriction.</p>
130       * 
131       * @param value the new value; must not be {@code null}
132       *
133       * @exception IllegalStateException if this method is called more
134       * than once
135       */
136      public void setValue(final String value) {
137        if (value == null) {
138          throw new IllegalArgumentException("value", new NullPointerException("value"));
139        }
140        final Object old = this.getValue();
141        if (old != null && !old.equals(value)) {
142          throw new IllegalStateException();
143        }
144        this.value = value;
145      }
146    
147      /**
148       * Returns a hashcode for this {@link AbstractValued} based off the
149       * hashcode of the return value of its {@link #getValue()} method.
150       *
151       * @return a hashcode for this {@link AbstractValued} implementation
152       */
153      @Override
154      public int hashCode() {
155        final int hashCode;
156        final String value = this.getValue();
157        if (value == null) {
158          hashCode = 0;
159        } else {
160          hashCode = value.hashCode();
161        }
162        return hashCode;
163      }
164    
165      /**
166       * Returns {@code true} if the supplied {@link Object} is an
167       * instance of {@code Valued} and returns a value from its {@link
168       * #getValue()} method that is {@linkplain String#equals(Object)
169       * equal to} the return value of this {@link AbstractValued}'s
170       * {@link #getValue()} method.
171       *
172       * @return {@code true} if the supplied {@link Object} is equal to
173       * this {@link AbstractValued}; {@code false} in all other cases
174       *
175       * @see #getValue()
176       */
177      @Override
178      public boolean equals(final Object other) {
179        if (other == this) {
180          return true;
181        } else if (other instanceof Valued) {
182          final Valued him = (Valued)other;
183          final Object value = this.getValue();
184          if (value == null) {
185            if (him.getValue() != null) {
186              return false;
187            }
188          } else if (!value.equals(him.getValue())) {
189            return false;
190          }
191          return true;
192        } else {
193          return false;
194        }
195      }
196    
197      /**
198       * Returns the return value of invoking {@link
199       * String#valueOf(Object)} on the return value of the {@link
200       * #getValue()} method.
201       *
202       * <p>This method never returns {@code null}.</p>
203       *
204       * @return the return value of invoking {@link
205       * String#valueOf(Object)} on the return value of the {@link
206       * #getValue()} method; never {@code null}
207       */
208      @Override
209      public String toString() {
210        return String.valueOf(this.getValue());
211      }
212    
213    }
