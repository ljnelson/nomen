028package com.edugility.nomen;
029
030import java.io.Serializable;
031
032/**
033 * A {@link Serializable} {@link Valued} implementation that correctly
034 * implements the {@link #equals(Object)} and {@link #hashCode()}
035 * methods.
036 *
037 * @author <a href="http://about.me/lairdnelson"
038 * target="_parent">Laird Nelson</a>
039 *
040 * @see #getValue()
041 */
042public abstract class AbstractValued implements Serializable, Valued {
043
044
045  /*
046   * Static fields.
047   */
048
049
050  /**
051   * The version of this class for {@linkplain Serializable
052   * serialization purposes}.
053   *
054   * @see Serializable
055   */
056  private static final long serialVersionUID = 1L;
057
058
059  /*
060   * Instance fields.
061   */
062
063
064  /**
065   * The value of this {@link AbstractValued}.
066   *
067   * <p>This field may be {@code null}.</p>
068   *
069   * @see #getValue()
070   *
071   * @see #setValue(String)
072   */
073  private String value;
074
075
076  /*
077   * Constructors.
078   */
079
080
081  /**
082   * Creates a new {@link AbstractValued} with no {@linkplain
083   * #getValue() value}.  The caller is expected to invoke the {@link
084   * #setValue(String)} method to complete initialization.
085   *
086   * @see #setValue(String)
087   */
088  protected AbstractValued() {
089    super();
090  }
091
092  /**
093   * Creates a new {@link AbstractValued} with the supplied {@code
094   * value}.
095   *
096   * @param value the new value; must not be {@code null}
097   *
098   * @exception IllegalArgumentException if {@code value} is {@code
099   * null}
100   *
101   * @see #setValue(String)
102   */
103  protected AbstractValued(final String value) {
104    this();
105    this.setValue(value);
106  }
107
108
109  /*
110   * Instance methods.
111   */
112
113
114  /**
115   * Returns the value of this {@link AbstractValued}.
116   *
117   * <p>This method may return {@code null}.</p>
118   *
119   * @return the value of this {@link AbstractValued}, or {@code
120   * null}
121   *
122   * @see #setValue(String)
123   *
124   * @see #AbstractValued(String)
125   */
126  @Override
127  public String getValue() {
128    return this.value;
129  }
130
131  /**
132   * Sets the value of this {@link AbstractValued}.
133   *
134   * <p>The default implementation of this method may only be called
135   * once, whether by {@linkplain #AbstractValued(String) an
136   * appropriate constructor} or afterwards.  Subsequent calls will
137   * throw an {@link IllegalStateException}.  Subclasses are free to
138   * override this restriction.</p>
139   * 
140   * @param value the new value; must not be {@code null}
141   *
142   * @exception IllegalStateException if this method is called more
143   * than once
144   *
145   * @see #AbstractValued(String)
146   */
147  public void setValue(final String value) {
148    if (value == null) {
149      throw new IllegalArgumentException("value", new NullPointerException("value"));
150    }
151    final Object old = this.getValue();
152    if (old != null && !old.equals(value)) {
153      throw new IllegalStateException();
154    }
155    this.value = value;
156  }
157
158  /**
159   * Returns a hashcode for this {@link AbstractValued} based off the
160   * hashcode of the return value of its {@link #getValue()} method.
161   *
162   * @return a hashcode for this {@link AbstractValued} implementation
163   */
164  @Override
165  public int hashCode() {
166    final int hashCode;
167    final String value = this.getValue();
168    if (value == null) {
169      hashCode = 0;
170    } else {
171      hashCode = value.hashCode();
172    }
173    return hashCode;
174  }
175
176  /**
177   * Returns {@code true} if the supplied {@link Object} is an
178   * instance of {@code Valued} and returns a value from its {@link
179   * #getValue()} method that is {@linkplain String#equals(Object)
180   * equal to} the return value of this {@link AbstractValued}'s
181   * {@link #getValue()} method.
182   *
183   * @return {@code true} if the supplied {@link Object} is equal to
184   * this {@link AbstractValued}; {@code false} in all other cases
185   *
186   * @see #getValue()
187   */
188  @Override
189  public boolean equals(final Object other) {
190    if (other == this) {
191      return true;
192    } else if (other instanceof Valued) {
193      final Valued him = (Valued)other;
194      final Object value = this.getValue();
195      if (value == null) {
196        if (him.getValue() != null) {
197          return false;
198        }
199      } else if (!value.equals(him.getValue())) {
200        return false;
201      }
202      return true;
203    } else {
204      return false;
205    }
206  }
207
208  /**
209   * Returns the return value of invoking {@link
210   * String#valueOf(Object)} on the return value of the {@link
211   * #getValue()} method.
212   *
213   * <p>This method never returns {@code null}.</p>
214   *
215   * @return the return value of invoking {@link
216   * String#valueOf(Object)} on the return value of the {@link
217   * #getValue()} method; never {@code null}
218   *
219   * @see String#valueOf(Object)
220   */
221  @Override
222  public String toString() {
223    return String.valueOf(this.getValue());
224  }
225
226}
